71 void JediManager::loadConfiguration(Lua* luaEngine) {
72  luaEngine->runFile("scripts/managers/jedi_manager.lua");
73 
74  jediProgressionType = luaEngine->getGlobalInt(String("jediProgressionType"));
75 
76  switch (jediProgressionType) {
77  case HOLOCRONJEDIPROGRESSION:
78  luaEngine->runFile("scripts/managers/holocron_jedi_manager.lua");
79  break;
80  case VILLAGEJEDIPROGRESSION:
81  luaEngine->runFile("scripts/managers/village_jedi_manager.lua");
82  break;
83  case CUSTOMJEDIPROGRESSION:
84  luaEngine->runFile(luaEngine->getGlobalString(String("customJediProgressionFile")));
85  break;
86  default:
87  break;
88  }
89 
90  setJediManagerName(luaEngine->getGlobalString(String("jediManagerName")));
91 
92  info("Loaded.", true);
93 }
