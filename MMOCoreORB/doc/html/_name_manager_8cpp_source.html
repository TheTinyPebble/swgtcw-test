191 void NameManager::fillNames() {
192  File* restrictedFile = new File("conf/restrictednames.lst");
193 
194  try {
195  FileReader restrictedReader(restrictedFile);
196 
197  info("parsing restricted names list: restrictednames.lst", true);
198 
199  BannedNameSet* setp;
200 
201  String line;
202  bool isset = false;
203 
204  while (restrictedReader.readLine(line)) {
205  String name = line.trim().toLowerCase();
206 
207  if ((name.length() >= 2 && name.subString(0, 2).compareTo("--") == 0)
208  || name == "") {
209  continue; //skip it
210  } else if (name.indexOf("[profane]") != -1) {
211  isset = false;
212  continue;
213  } else if (name.indexOf("[developer]") != -1) {
214  isset = true;
215  setp = developerNames;
216  continue;
217  } else if (name.indexOf("[fiction]") != -1) {
218  isset = true;
219  setp = fictionNames;
220  continue;
221  } else if (name.indexOf("[reserved]") != -1) {
222  isset = true;
223  setp = reservedNames;
224  continue;
225  } else if (isset) {
226  setp->add(name);
227  continue;
228  } else {
229  profaneNames->add(name);
230  continue;
231  }
232  }
233  } catch (FileNotFoundException&e ) {
234  }
235 
236  delete restrictedFile;
237 }

273 int NameManager::validateName(const String& name, int species) {
274  if (name.isEmpty())
275  return NameManagerResult::DECLINED_EMPTY;
276 
277  if (name.length() < 3)
278  return NameManagerResult::DECLINED_RACE_INAPP;
279 
280  if (isProfane(name))
281  return NameManagerResult::DECLINED_PROFANE;
282 
283  if (isDeveloper(name))
284  return NameManagerResult::DECLINED_DEVELOPER;
285 
286  if (isFiction(name))
287  return NameManagerResult::DECLINED_FICT_RESERVED;
288 
289  if (isReserved(name))
290  return NameManagerResult::DECLINED_RESERVED;
291 
292  //Make sure that only valid characters are allowed in the name.
293  if (strspn(name.toCharArray(), "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz'- ") != name.length())
294  return NameManagerResult::DECLINED_SYNTAX;
295 
296  String fname, lname;
297 
298  //Split the name into first and last
299  int spc = name.indexOf(" ");
300  if (spc != -1) {
301  fname = name.subString(0, spc);
302  lname = name.subString(spc + 1, name.length());
303  } else {
304  fname = name;
305  lname = "";
306  }
307 
308  if (fname.length() < 3 || fname.length() > 15 || lname.length() > 20)
309  return NameManagerResult::DECLINED_RACE_INAPP;
310 
311  //Wookies are not allowed to have last names.
312  if (!lname.isEmpty() && species == CreatureObject::WOOKIE)
313  return NameManagerResult::DECLINED_RACE_INAPP;
314 
315  //If the name has a hyphen or apostrophe, make sure they are the proper species.
316  if (name.indexOf("'") != -1 || name.indexOf("-") != -1) {
317  //Must be a human, twilek, or moncal to have a hyphen or apostrophe.
318  if (species != CreatureObjectImplementation::HUMAN && species != CreatureObjectImplementation::TWILEK && species != CreatureObjectImplementation::MONCAL && species != CreatureObjectImplementation::ZABRAK)
319  return NameManagerResult::DECLINED_RACE_INAPP;
320 
321  //Moncal's aren't allowed to have apostrophes.
322  if (species == CreatureObjectImplementation::MONCAL && name.indexOf("'") != -1)
323  return NameManagerResult::DECLINED_RACE_INAPP;
324 
325  //Make sure they only have one hyphen and apostrophe in firstname.
326  if (fname.indexOf('\'') != fname.lastIndexOf('\'') || fname.indexOf('-') != fname.lastIndexOf('-'))
327  return NameManagerResult::DECLINED_RACE_INAPP;
328 
329  //Make sure they only have one hyphen and apostrophe in lastname.
330  if (lname.indexOf('\'') != lname.lastIndexOf('\'') || lname.indexOf('-') != lname.lastIndexOf('-'))
331  return NameManagerResult::DECLINED_RACE_INAPP;
332  }
333 
334  return NameManagerResult::ACCEPTED;
335 }

349 String NameManager::makeName(int nameLength) {
350  String name;
351 
352  do {
353  name = "";
354 
356  name += chooseNextLetter(' ', ' ');
357  name += chooseNextLetter(name[0], ' ');
358 
359  assert(name.length() != 0);
360 
361  while (name.length() < nameLength)
362  name += chooseNextLetter(name[name.length() - 1], name[name.length() - 2]);
363 
364  name[0] = Character::toUpperCase(name[0]);
365 
366  } while (validateName(name, -1) != NameManagerResult::ACCEPTED);
367 
368  return name;

402 char NameManager::chooseNextLetter(const char lastLetter, const char letterBeforeLast) {
403 
404  if (letterBeforeLast == ' ' && lastLetter == ' ')
405  return 97 + System::random(25);
406 
407  StringBuffer last;
408  last << Character::toLowerCase(lastLetter);
409 
410  String inclusion = "abcdefghijklmnopqrstuvwxyz";
411 
412  if (letterBeforeLast != ' ' && (!isVowel(lastLetter) && !isVowel(letterBeforeLast))) {
413 
414  inclusion = letterMappings.get("doubleconsonent");
415 
416  } else if ((letterBeforeLast != ' ' && isVowel(lastLetter) && isVowel(letterBeforeLast))) {
417 
418  inclusion = letterMappings.get("doublevowel");
419 
420  } else if ((letterBeforeLast != ' ' && lastLetter == 'u' && letterBeforeLast == 'q')) {
421 
422  inclusion = letterMappings.get("qu");
423 
424  } else if(letterMappings.contains(last.toString())) {
425 
426  inclusion = letterMappings.get(last.toString());
427  }
428 
429  return chooseLetterInclusive(inclusion);
430 }
