204 Vector<String> ResourceSpawnImplementation::getSpawnZones(int minpool, int maxpool,
205  const String& zonerestriction, Vector<String>& activeZones) {
206 
211  Vector<String> zonenames;
212  int zonecount = 0;
213 
214  if(minpool == maxpool)
215  zonecount = maxpool;
216  else
217  zonecount = System::random(maxpool - minpool) + minpool;
218 
219  if(zonecount > activeZones.size())
220  zonecount = activeZones.size();
221 
223  if (zonerestriction != "") {
224  zonenames.add(zonerestriction);
225  return zonenames;
226  }
227 
230  while (activeZones.size() > zonecount)
231  activeZones.remove(System::random(activeZones.size() - 1));
232 
233  return activeZones;
234 }
