122  byte getWeatherAt(float x, float y) {
123 
124  if(time(0) > endTime)
125  return 0;
126 
127 
128  float mid = (endTime - startTime) / 2.f;
129  float current = abs(time(0) - startTime);
130  float end = endTime - startTime;
131 
132  if(current > mid)
133  current = end - current;
134 
135  float currentDensity = (current / mid) * density;
136  x -= minX;
137  y = maxY - y;
138 
139  // [0,1]
140  float value = (SimplexNoise::noise(x * modifier, y * modifier, seed * modifier) + 1) / 2.0f;
141 
142  byte weather = (byte)(((value * currentDensity) * 100) / 20.f);
