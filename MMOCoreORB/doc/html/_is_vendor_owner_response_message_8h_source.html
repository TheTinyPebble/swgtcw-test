71 
72  if(vendor->isVendor()) {
73 
74 
75  DataObjectComponentReference* data = vendor->getDataObjectComponent();

77  return;
78  }
79 
80  VendorDataComponent* vendorData = cast<VendorDataComponent*>(data->get());
81  if(vendorData == NULL) {
82  return;
83  }
84 
85  rights = vendorData->getOwnershipRightsOf(player);
86 
87  if(vendorData->isOnStrike()) {
88  marketIssues = true;
89  player->sendSystemMessage("@ui_auc:err_vendor_deactivated");
90  }
91  }
92 
93  uint64 objectID = vendor->getObjectID();
94 
95  // 0: own vendor, 1: someone else owns the vendor, 2: the galaxy owns the vendor (bazaar)
96  insertInt(rights);
97 
98  insertInt(marketIssues);
99 
100  insertLong(objectID);
101 
102  StringBuffer title;

104 
105  int x = vendor->getWorldPositionX();

108  String uuid = auctionManager->getVendorUID(vendor);

110  insertAscii(uuid);

112  insertShort(0x64); // ?? 64
