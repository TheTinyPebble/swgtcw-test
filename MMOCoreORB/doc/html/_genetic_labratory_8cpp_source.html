18 float GeneticLabratory::applyFormula(float aVal, float bVal, float cVal, float dVal, float eVal,int formula) {
19  float a = aVal;
20  float b = bVal;
21  float c = cVal;
22  float d = dVal;
23  float e = eVal;
24  float rc = 0;
25  if (aVal < 0)
26  a = -99;
27  if (bVal < 0)
28  b = -99;
29  if (cVal < 0)
30  c = -99;
31  if (dVal < 0)
32  d = -99;
33  if (eVal < 0)
34  e = -99;
35  switch (formula) {
36  case PHYSIQUE:
37  rc = (a * 0.4) + (b *0.25) + (c * 0.05) + (d * 0.05) + (e * 0.25);
38  break;
39  case PROWESS:
40  rc = (a * 0.25) + (b *0.42) + (c * 0.17) + (d * 0.085) + (e * .075);
41  break;
42  case MENTAL:
43  rc = (a * 0.05) + (b *0.1) + (c * 0.5) + (d * 0.3) + (e * .05);
44  break;
45  case PHYSCHOLOGICAL:
46  rc = (a * 0.09) + (b *0.05) + (c * 0.26) + (d * 0.43) + (e * 0.17);
47  break;
48  case AGRESSION:
49  rc = (a * 0.17) + (b *0.16) + (c * 0.085) + (d * 0.165) + (e * 0.42);
50  break;
51  }
52  if (rc > 1000)
53  rc = 1000;
54  return rc;
55 }
