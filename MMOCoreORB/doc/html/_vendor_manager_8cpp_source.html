156 String VendorManager::getTimeString(uint32 timestamp) {
157 
158  if( timestamp == 0 ){
159  return "";
160  }
161 
162  String abbrvs[3] = { "minutes", "hours", "days" };
163 
164  int intervals[3] = { 60, 3600, 86400 };
165  int values[3] = { 0, 0, 0 };
166 
167  StringBuffer str;
168 
169  for (int i = 2; i > -1; --i) {
170  values[i] = floor((float) timestamp / intervals[i]);
171  timestamp -= values[i] * intervals[i];
172 
173  if (values[i] > 0) {
174  if (str.length() > 0){
175  str << ", ";
176  }
177 
178  str << values[i] << " " << abbrvs[i];
179  }
180  }
181 
182  return "(" + str.toString() + ")";
183 }
