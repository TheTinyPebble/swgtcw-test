74 float VisibilityManager::calculateVisibilityIncrease(CreatureObject* creature) {
75  ManagedReference<Zone*> zone = creature->getZone();
76 
77  float visibilityIncrease = 0;
78 
79  if (zone != NULL) {
80  SortedVector<ManagedReference<QuadTreeEntry* > > closeObjects;
81  zone->getInRangeObjects(creature->getWorldPositionX(), creature->getWorldPositionY(), 32, &closeObjects, true);
82 
83  for (int i = 0; i < closeObjects.size(); ++i) {
84  SceneObject* obj = cast<SceneObject*>(closeObjects.get(i).get());
85  if (obj->isCreatureObject()) {
86  ManagedReference<CreatureObject*> c = cast<CreatureObject*>(obj);
87  if (c->isNonPlayerCreatureObject()) {
88  if (creature->getFaction() == 0 || (c->getFaction() != factionImperial && c->getFaction() != factionRebel)) {
89  visibilityIncrease += 0.5;
90  } else {
91  if (creature->getFaction() == c->getFaction()) {
92  visibilityIncrease += 0.25;
93  } else {
94  visibilityIncrease += 1;
95  }
96  }
97  }
98  }
99  }
100  }
101 
102  //info("Increasing visibility for player " + String::valueOf(creature->getObjectID()) + " with " + String::valueOf(visibilityIncrease), true);
103  return visibilityIncrease;
104 }

106 void VisibilityManager::decreaseVisibility(CreatureObject* creature) {
107  Reference<PlayerObject*> ghost = creature->getSlottedObject("ghost").castTo<PlayerObject*>();
108 
109  if (ghost != NULL) {
110  Locker locker(ghost);
111  if (ghost->getVisibility() > 0)
112  {
113  float visibilityDecrease = ghost->getLastVisibilityUpdateTimestamp().miliDifference() / 1000.f / 3600.f;
114  if (ghost->getVisibility() <= visibilityDecrease) {
115  clearVisibility(creature);
116  } else {
117  ghost->setVisibility(ghost->getVisibility() - visibilityDecrease);
118 
119  if (ghost->getVisibility() < TERMINALVISIBILITYLIMIT) {
120  removePlayerFromBountyList(creature);
121  }
122  }
123  }
124  }
125 }

132 void VisibilityManager::login(CreatureObject* creature) {
133  //info("Logging in " + creature->getFirstName(), true);
134  Reference<PlayerObject*> ghost = creature->getSlottedObject("ghost").castTo<PlayerObject*>();
135 
136  if (ghost != NULL) {
137  decreaseVisibility(creature);
138 
139  Locker locker(&visibilityListLock);
140 
141  if ((ghost->getVisibility() > 0) && (!visibilityList.contains(creature->getObjectID()))) {
142  //info("Adding player " + String::valueOf(creature->getObjectID()) + " to visibility list.", true);
143  visibilityList.put(creature->getObjectID(), creature);
144  }
145 
146  locker.release();
147 
148  if (ghost->getVisibility() >= TERMINALVISIBILITYLIMIT) {
149  addPlayerToBountyList(creature, calculateReward(ghost->getVisibility()));
150  }
151  }
152 }
