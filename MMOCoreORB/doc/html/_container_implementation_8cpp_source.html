161  if ((object->isIntangibleObject() && getContainerType() != 3)

163  errorDescription = "@container_error_message:container07"; // You cannot put that kind of item in that kind of container.
164 
165  return TransferErrorCode::INVALIDTYPE;
166  }
167 
168  if (containmentType == -1) {
169  if ((gameObjectType == SceneObjectType::WEARABLECONTAINER && object->getGameObjectType() == SceneObjectType::WEARABLECONTAINER)) {
170  errorDescription = "@container_error_message:container12"; // This item is too bulky to fit inside this container.
171

181  // Find out how much room we need
182  int objectSize;
183 
184  if (object->isContainerObject())
185  objectSize = object->getContainerObjectsSize() + 1;
186  else
187  objectSize = 1;
188 
189  // Return if there's not enough room in the container
190  if (getContainerVolumeLimit() < getCountableObjectsRecursive() + objectSize) {
191  errorDescription = "@container_error_message:container03"; // This container is full.

199  // If there's a wearable container parent, return if it doesn't have enough room
200  if (wearableParent != NULL) {

202  errorDescription = "@container_error_message:container03"; // This container is full.

217  errorDescription = "@error_message:inv_full"; // Your inventory is full.
218 
219  return TransferErrorCode::CONTAINERFULL;
220  }
221  // Return if it's in a player bank that doesn't have room
222  } else if (parentOfWearableParent == bank) {
223  if (bank->getContainerVolumeLimit() < bank->getCountableObjectsRecursive() + objectSize) {
224  errorDescription = "@container_error_message:container03"; // This container is full.
225 
226  return TransferErrorCode::CONTAINERFULL;
227  }
228  }
229  }
230  } else {
231  // There's no parent that's a wearable container. Check if this is
232  if (gameObjectType == SceneObjectType::WEARABLECONTAINER) {
233  WearableContainerObject* pack = cast<WearableContainerObject*>(_this.get().get());
234 
235  if (pack != NULL && !pack->isEquipped()) {
236  // This is a wearable container, and it's not equipped.
237  if (playerParent != NULL ) {

239  SceneObject* bank = playerParent->getSlottedObject("bank");

241

263  SceneObject* bank = playerParent->getSlottedObject("bank");
264  SceneObject* thisParent = getParent().get();

289  if (rootParent->isBuildingObject()) {

304  }
305  }
306  }
307  }
