134  int doCombatAction(CreatureObject* creature, const uint64& target, const UnicodeString& arguments = "", ManagedReference<WeaponObject*> weapon = NULL) {
135  ManagedReference<SceneObject*> targetObject = server->getZoneServer()->getObject(target);
136  PlayerManager* playerManager = server->getPlayerManager();
137 
138  if (targetObject == NULL || !targetObject->isTangibleObject() || targetObject == creature)
139  return INVALIDTARGET;
140 
141  float checkRange = range;
142 
143  if (weapon == NULL) {
144  if(creature->getWeapon() == NULL) {
145  return GENERALERROR;
146  }
147  else {
148  weapon = creature->getWeapon();
149  }
150  }
151 
152  if (checkRange == -1) {
153  checkRange = weapon->getMaxRange();
154  }
155 
156  if (creature->isDead())
157  return INVALIDLOCOMOTION;
158 
159  if (creature->isPlayerCreature()){
160  if (creature->getPlayerObject() && creature->getPlayerObject()->isAFK()) {
161  return GENERALERROR;
162  }
163  }
164  if (creature->isKneeling() && weapon->isMeleeWeapon() && !weapon->isJediWeapon())
165  return NOKNEELING;
166 
167  if (creature->isProne() && !weapon->isRangedWeapon())
168  return NOPRONE;
169 
170  if (!targetObject->isInRange(creature, checkRange))
171  return TOOFAR;
172 
173  if (!CollisionManager::checkLineOfSight(creature, targetObject)) {
174  creature->sendSystemMessage("@container_error_message:container18");
175  return GENERALERROR;
176  }
177 
178  CombatManager* combatManager = CombatManager::instance();
179 
180  try {
181  int res = combatManager->doCombatAction(creature, weapon, cast<TangibleObject*>(targetObject.get()), CreatureAttackData(arguments, this));
182 
183  switch (res) {
184  case -1:
185  return INVALIDTARGET;
186  case -2:
187  return INSUFFICIENTHAM;
188  case -3:
189  return GENERALERROR;
190  }
191 
192  } catch (Exception& e) {
193  error("unreported exception caught in CombatQueueCommand::doCombatAction");
194  error(e.getMessage());
195  e.printStackTrace();
196  }
197 
198  // only clear aiming states if command was successful
199  creature->removeStateBuff(CreatureState::AIMING);
200  creature->removeBuff(String("steadyaim").hashCode());
201 
202  return SUCCESS;
203  }

534  virtual void applyEffect(CreatureObject* creature, uint8 effectType, uint32 mod) {
535  CombatManager* combatManager = CombatManager::instance();
536  StateEffect effect = getStateEffect(effectType);
537  Reference<Buff*> buff = NULL;
538 
539  switch (effectType) {
540  case CommandEffect::BLIND:
541  creature->setBlindedState(effect.getStateLength());
542  break;
543  case CommandEffect::DIZZY:
544  creature->setDizziedState(effect.getStateLength());
545  break;
546  case CommandEffect::INTIMIDATE:
547  creature->setIntimidatedState(mod, effect.getStateLength());
548  break;
549  case CommandEffect::STUN:
550  creature->setStunnedState(effect.getStateLength());
551  break;
552  case CommandEffect::KNOCKDOWN:
553  if (creature->isKnockedDown() || creature->isProne()) {
554  if (80 > System::random(100))
555  creature->setPosture(CreaturePosture::UPRIGHT, true);
556  break;
557  }
558 
559  if (creature->isMounted()) {
560  creature->updateCooldownTimer("mount_dismount", 0);
561  creature->dismount();
562  }
563 
564  if (!creature->isDead() && !creature->isIncapacitated())
565  creature->setPosture(CreaturePosture::KNOCKEDDOWN);
566 
567  creature->updateKnockdownRecovery();
568  creature->updateLastKnockdown();
569  creature->sendSystemMessage("@cbt_spam:posture_knocked_down");
570 
571  break;
572  case CommandEffect::POSTUREUP:
573  if (creature->isMounted()) {
574  creature->updateCooldownTimer("mount_dismount", 0);
575  creature->dismount();
576  }
577 
578  if (creature->getPosture() == CreaturePosture::PRONE) {
579  creature->setPosture(CreaturePosture::CROUCHED);
580  creature->sendSystemMessage("@cbt_spam:force_posture_change_1");
581  creature->updatePostureUpRecovery();
582  } else if (creature->getPosture() == CreaturePosture::CROUCHED) {
583  creature->setPosture(CreaturePosture::UPRIGHT);
584  creature->sendSystemMessage("@cbt_spam:force_posture_change_0");
585  creature->updatePostureUpRecovery();
586  }
587 
588  break;
589  case CommandEffect::POSTUREDOWN:
590  if (creature->isMounted()) {
591  creature->updateCooldownTimer("mount_dismount", 0);
592  creature->dismount();
593  }
594 
595  if (creature->getPosture() == CreaturePosture::UPRIGHT) {
596  creature->setPosture(CreaturePosture::CROUCHED);
597  creature->sendSystemMessage("@cbt_spam:force_posture_change_1");
598  creature->updatePostureDownRecovery();
599  } else if (creature->getPosture() == CreaturePosture::CROUCHED) {
600  creature->setPosture(CreaturePosture::PRONE);
601  creature->sendSystemMessage("@cbt_spam:force_posture_change_2");
602  creature->updatePostureDownRecovery();
603  }
604 
605  break;
606  case CommandEffect::NEXTATTACKDELAY:
607  creature->setNextAttackDelay(mod, effect.getStateLength());
608  break;
609  case CommandEffect::HEALTHDEGRADE:
610  buff = new Buff(creature, String("healthdegrade").hashCode(), effect.getStateLength(), BuffType::STATE);
611  buff->setAttributeModifier(CreatureAttribute::CONSTITUTION, -1*effect.getStateStrength());
612  buff->setAttributeModifier(CreatureAttribute::STRENGTH, -1*effect.getStateStrength());
613  creature->addBuff(buff);

617  buff->setAttributeModifier(CreatureAttribute::QUICKNESS, -1*effect.getStateStrength());

619  creature->addBuff(buff);
