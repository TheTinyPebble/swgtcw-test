338 bool ForageManagerImplementation::forageGiveResource(CreatureObject* player, float forageX, float forageY, const String& planet, String& resType) {
339  if (player == NULL)

348 
349  if(resType.isEmpty()) {
350  //Get a list of the flora on the planet.

353  if (resources.size() < 1)
354  return false;
355 
356  while (resources.size() > 0) {
357  int key = System::random(resources.size() - 1);
358  float density = resources.get(key)->getDensityAt(planet, forageX, forageY);
359 
360  if (density <= 0.0 && resources.size() > 1) { //No concentration of this resource near the player.
361  resources.remove(key); //Remove and pick another one.
362 
363  } else { //If there is only one left, we give them that one even if density is 0.
364  resource = resources.get(key);
365  break;
366  }
367  }
368  } else {
369  if(player->getZone() == NULL)
370  return false;
371 
372  resType = resType + "_" + player->getZone()->getZoneName();
373  resource = resourceManager->getCurrentSpawn(resType, player->getZone()->getZoneName());
374 
375  if(resource == NULL) {
376  StringBuffer message;
377  message << "Resource type not available: " << resType << " on " << player->getZone()->getZoneName();
378  warning(message.toString());
379  return false;
380  }
381  }
382 
383  int quantity = System::random(30) + 10;
384  resourceManager->harvestResourceToPlayer(player, resource, quantity);
385  return true;
