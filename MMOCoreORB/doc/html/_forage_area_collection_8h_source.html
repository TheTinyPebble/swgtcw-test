61  ForageAreaCollection(CreatureObject* player, short forageX, short forageY, const String& planet) {
62  //Player is foraging for the first time, so make a new forage area.
63  Reference<ForageArea*> newArea = new ForageArea(forageX, forageY, planet);
64  forageAreas.add(newArea);

68  cleanupEvent->schedule(1800000); //30 minutes

72  bool checkForageAreas(short forageX, short forageY, const String& planet) {
73  Reference<ForageArea*> forageArea;
74  int8 zoneStatus = -1;
75  int8 authorized = -1;
76 
77  //Reschedule the cleanup event.
78  cleanupEvent->reschedule(1800000); //30 minutes
79 
80  //Check forage areas for permission to forage.
81  for (int i = 0; i < forageAreas.size(); i++) {
82  forageArea = forageAreas.get(i);
83  zoneStatus = forageArea->checkPermission(forageX, forageY, planet);
84 
85  switch (zoneStatus) {
86 
87  case 0: //Zone is time expired, delete.
88  forageAreas.remove(i);
89  i -= 1; //Indexes of remaining areas shift down 1 when deleting an element, so compensate.
90  break;
91 
92  case 1: //Player is not in area, but it's not expired either.
93  break;
94 
95  case 2: //Player is in this zone but was allowed to forage.
96  authorized = i; //Remember which zone gave permission.
97  break;
98 
99  case 3: //Player is not allowed to forage in this zone, deny.
100  return false;
101 
102  default: //Shouldn't happen.
103  return false;
104 
105  }
106  }
107 
108  //Check if a zone gave permission to forage.
109  if (authorized != -1) {
110  forageArea = forageAreas.get(authorized);
111  forageArea->uses += 1; //Add 1 use to the last zone to allow forage (zones can overlap).
112 
113  } else { //Player is not in any zones, so make a new one.
114  if (forageAreas.size() == 120) //Cap how many zones we remember.
115  forageAreas.remove(0); //If at capacity, delete the oldest zone.
116 
117  Reference<ForageArea*> newArea = new ForageArea(forageX, forageY, planet);
118  forageAreas.add(newArea);
119  }
120 
121  return true;
122  }
