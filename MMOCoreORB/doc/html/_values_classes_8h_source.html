149  inline void setValue(const float& value) {
150  if (locked)
151  return;
152 
153  float newpercentage;
154 
155  if (maxValue > minValue)
156  newpercentage = (value - minValue) / (maxValue - minValue);
157  else
158  newpercentage = 1 - ((value - maxValue) / (minValue - maxValue));
159 
160  if(newpercentage > values.get("maxPercentage"))
161  newpercentage = values.get("maxPercentage");
162 
163  if(newpercentage < 0)
164  newpercentage = 0;
165 
166  if (values.contains("currentValue")) {
167  values.drop("currentValue");
168  }
169 
170  if (values.contains("currentPercentage")) {
171  values.drop("currentPercentage");
172  }
173 
174  values.put("currentValue", value);
175  values.put("currentPercentage", newpercentage);
176  }

234  inline void resetValue() {
235  float reset = (getMaxPercentage() * 10.0f) * (0.000015f * (getMaxPercentage() * 10.0f) + 0.015f);
236  setPercentage(reset);

238  float newvalue;
