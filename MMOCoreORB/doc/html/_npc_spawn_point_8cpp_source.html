55 NpcSpawnPoint::NpcSpawnPoint(CreatureObject* player, const String& spawnTypes) {
56  //Parse spawn types.
57  String st = spawnTypes.toLowerCase();
58  spawnType = 0;
59  if (st.contains("neutral")) {
60  spawnType |= NEUTRALSPAWN;
61  }
62  if (st.contains("imperial")) {
63  spawnType |= IMPERIALSPAWN;
64  }
65  if (st.contains("rebel")) {
66  spawnType |= REBELSPAWN;
67  }
68  if (st.contains("bhtarget")) {
69  spawnType |= BHTARGETSPAWN;
70  }
71  if (st.contains("nospawn")) {
72  //No spawn overrides all other spawn types.
73  spawnType = NOSPAWN;
74  }
75  position.setX(player->getPosition().getX());
76  position.setY(player->getPosition().getY());
77  direction.setHeadingDirection(player->getDirection()->getRadians());
78  inUseByNumberOfMissions = 0;
79  despawnMissionNpcTask = NULL;
80  npcSpawned = false;
81 }

115 void NpcSpawnPoint::allocateNpc(TerrainManager* terrainManager, CreatureManager* creatureManager) {
116  inUseByNumberOfMissions++;
117 
118  if (inUseByNumberOfMissions > 0) {
119  if (despawnMissionNpcTask != NULL && despawnMissionNpcTask->isScheduled()) {
120  despawnMissionNpcTask->cancel();
121  }
122 
123  if (!npcSpawned) {
124  //Spawn the NPC.
125  String deliverNpc = "deliver_npc";
126  float z = terrainManager->getHeight(position.getX(), position.getY());
127  npc = cast<AiAgent*>(creatureManager->spawnCreature(deliverNpc.hashCode(), 0, position.getX(), z, position.getY(), 0));
128  if (npc != NULL) {
129  npc->updateDirection(direction.getW(), direction.getX(), direction.getY(), direction.getZ());
130  //Set the name of the NPC.
131  NameManager* nm = npc->getZoneProcessServer()->getNameManager();
132  npc->setCustomObjectName(nm->makeCreatureName(), true);
133  npcSpawned = true;
134  } else {
135  info("Failed to spawn npc at " + position.toString(), true);
136  }
137  }
138  }
139 }
