154 void StructureObjectImplementation::scheduleMaintenanceExpirationEvent() {
155  if (getMaintenanceRate() <= 0) {
156  //No maintenance cost, structure maintenance cannot expire.
157  return;
158  }
159 
160  int timeRemaining;
161 
162  if (structureMaintenanceTask != NULL) {
163  updateStructureStatus();
164 
165  float cityTax = 0.f;
166 
167  ManagedReference<CityRegion*> city = _this.get()->getCityRegion();
168 
169  if(city != NULL) {
170  cityTax = city->getPropertyTax();
171  }
172 
173  timeRemaining = (int) (surplusMaintenance * 3600.f / (getMaintenanceRate() + (getMaintenanceRate() * cityTax / 100) ));
174 
175  if (timeRemaining <= 0) {
176  //Decaying structures should be scheduled as soon as possible. Maintenance task will handle
177  //any further rescheduling.
178  timeRemaining = 1;
179  } else if (timeRemaining > 24 * 60 * 60 * 1000) {
180  //Run maintenance task at least one time every day but randomized to spread it out.
181  timeRemaining = 12 * 60 * 60 + System::random(12 * 60 * 60);
182  }
183 
184  maintenanceExpires.updateToCurrentTime();
185  maintenanceExpires.addMiliTime(timeRemaining * 1000);
186  }
187  else
188  {
189  updateStructureStatus();
190 
191  timeRemaining = (int) (surplusMaintenance * 3600.f / getMaintenanceRate());
192 
193  if (timeRemaining <= 0) {
194  //Decaying structures should be scheduled as soon as possible. Maintenance task will handle
195  //any further rescheduling.
196 
197  //Randomize maintenance tasks over the first hour after server restart.
198  timeRemaining = System::random(60 * 60);
199  } else if (timeRemaining > 24 * 60 * 60 * 1000) {
200  //Run maintenance task at least one time every day but randomized to spread it out.
201  timeRemaining = 12 * 60 * 60 + System::random(12 * 60 * 60);
202  }
203 
204  maintenanceExpires.updateToCurrentTime();
205  maintenanceExpires.addMiliTime(timeRemaining * 1000);
206  }
207 
208  scheduleMaintenanceTask(timeRemaining);
209 }

251 void StructureObjectImplementation::updateStructureStatus() {
259  if(isCivicStructure())
260  return;
261 
262  float timeDiff = ((float) lastMaintenanceTime.miliDifference()) / 1000.f;
263  float maintenanceDue = (getMaintenanceRate() / 3600.f) * timeDiff;
264  float cityTaxDue = 0;
265 
266  if (maintenanceDue > 0) {
267  //Only update last time if we actually progressed to get correct consumption.
268  lastMaintenanceTime.updateToCurrentTime();
269  }
270 
271  ManagedReference<CityRegion*> city = getCityRegion();
272 
273  if(isBuildingObject() && city != NULL && !city->isClientRegion() && city->getPropertyTax() > 0){
274  cityTaxDue = city->getPropertyTax() / 100.0f * maintenanceDue;
275 
276  // sometimes a creature and building will be locked here
277  if(cityTaxDue > 0){
278  Reference<PayPropertyTaxTask*> taxTask = new PayPropertyTaxTask(city, cityTaxDue);
279  taxTask->execute();
280  }
281 
282  }
283 
284  //Maintenance is used as decay as well so let it go below 0.
285  surplusMaintenance -= ( maintenanceDue + cityTaxDue );
286 
287  //Update structure condition.
288  if (surplusMaintenance < 0) {
289  setConditionDamage(-surplusMaintenance, true);
290  } else {
291  setConditionDamage(0, true);

299 int StructureObjectImplementation::getDecayPercentage() {
300  //Update structure status.
301  updateStructureStatus();
302 
303  //Calculate decay.
304  int decayedBelowMaintenance = getMaxCondition();
305 
306  if (decayedBelowMaintenance != 0) {
307  int percentage = (decayedBelowMaintenance - getConditionDamage()) * 100 / decayedBelowMaintenance;
308 
309  //Calculation above truncates the percentage value but we want to round it upwards.
310  percentage += 1;
311  if (percentage < 0) {
312  percentage = 0;
313  } else if (percentage > 100) {
314  percentage = 100;
315  }
316 
317  return percentage;
318  } else {
319  //Structure has 0 in max condition, i.e. it cannot decay. Condition is therefore always 100 %.
320  return 100;
321  }
