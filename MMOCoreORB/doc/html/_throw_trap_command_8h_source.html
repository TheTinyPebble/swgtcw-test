60  int doQueueCommand(CreatureObject* creature, const uint64& target,
61  const UnicodeString& arguments) {
62 
63  if (!checkStateMask(creature))
64  return INVALIDSTATE;
65 
66  if (!checkInvalidLocomotions(creature))
67  return INVALIDLOCOMOTION;
68 
69  int skillLevel = creature->getSkillMod("trapping");
70  if (skillLevel < 1 || !creature->hasSkill("outdoors_scout_novice")) {
71  creature->sendSystemMessage("@trap/trap:trap_no_skill");
72  return GENERALERROR;
73  }
74 
75  StringTokenizer tokenizer(arguments.toString());
76 
77  if (!tokenizer.hasMoreTokens())
78  return INVALIDPARAMETERS;
79 
80  try {
81 
82  uint64 trapId = tokenizer.getLongToken();
83  ManagedReference<TangibleObject*> trap =
84  server->getZoneServer()->getObject(trapId).castTo<TangibleObject*>();
85 
86  if (trap == NULL)
87  return INVALIDPARAMETERS;
88 
89  if (!trap->isTrapObject())
90  return INVALIDPARAMETERS;
91 
92  if (!trap->isASubChildOf(creature))
93  return GENERALERROR;
94 
95  ManagedReference<CreatureObject*> targetCreature =
96  server->getZoneServer()->getObject(target).castTo<CreatureObject*>();
97 
98  if (targetCreature == NULL || !targetCreature->isCreature()) {
99  creature->sendSystemMessage("@trap/trap:sys_creatures_only");
100  return GENERALERROR;
101  }
102 
103  if (!(targetCreature->getPvpStatusBitmask() & CreatureFlag::ATTACKABLE)) {
104  creature->sendSystemMessage("@trap/trap:sys_no_pets");
105  return GENERALERROR;
106  }
107 
108  SharedObjectTemplate* templateData =
109  TemplateManager::instance()->getTemplate(
110  trap->getServerObjectCRC());
111  if (templateData == NULL) {
112  error("No template for: " + String::valueOf(trap->getServerObjectCRC()));
113  return GENERALERROR;
114  }
115 
116  TrapTemplate* trapData = cast<TrapTemplate*> (templateData);
117  if (trapData == NULL) {
118  error("No TrapTemplate for: " + String::valueOf(trap->getServerObjectCRC()));
119  return GENERALERROR;
120  }
121 
123  if(creature->getDistanceTo(targetCreature) > trapData->getMaxRange())
124  {
125  StringIdChatParameter tooFar("cmd_err", "target_range_prose");
126  tooFar.setTO("Throw Trap");
127 
128  creature->sendSystemMessage(tooFar);
129  return GENERALERROR;
130  }
131 
132  int effectType = 0;
133 
134  // No skill Check
135  int trappingSkill = creature->getSkillMod("trapping");
136  if(trappingSkill < 1) {
137  creature->sendSystemMessage("@trap/trap:trap_no_skill");
138  return GENERALERROR;
139  }
140 
142  if(trappingSkill < trapData->getSkillRequired()) {
143  creature->sendSystemMessage("@trap/trap:trap_no_skill_this");
144  return GENERALERROR;
145  }
146 
147  int targetDefense = targetCreature->getSkillMod(trapData->getDefenseMod());
148  Time* cooldown = creature->getCooldownTime("throwtrap");
149  if((cooldown != NULL && !cooldown->isPast()) ||
150  creature->getPendingTask("throwtrap") != NULL) {
151  creature->sendSystemMessage("@trap/trap:sys_not_ready");
152  return GENERALERROR;
153  }
154 
155  if(!CombatManager::instance()->startCombat(creature, targetCreature))
156  return GENERALERROR;
157 
158  float hitChance = CombatManager::instance()->hitChanceEquation(trappingSkill, 0, targetDefense);
159 
160  if (hitChance > 100)
161  hitChance = 100.0;
162  else if (hitChance < 0)
163  hitChance = 0;
164 
165  int roll = System::random(100);
166  uint64 state = trapData->getState();
167  bool hit = roll < hitChance && (state == 0 || (state != 0 && !targetCreature->hasState(state)));
168 
169  String animation = trapData->getAnimation();
170  uint32 crc = String(animation).hashCode();
171  CombatAction* action = new CombatAction(creature, targetCreature, crc, hit, 0L);
172  creature->broadcastMessage(action, true);
173  creature->addCooldown("throwtrap", 1500);
174 
175  trap->decreaseUseCount();
176 
177  StringIdChatParameter message;
178  ManagedReference<Buff*> buff = NULL;
179  int damage = 0;
180 
181  if (hit) {
182 
183  message.setStringId("trap/trap" , trapData->getSuccessMessage());
184 
185  buff = new Buff(targetCreature, crc, trapData->getDuration(), BuffType::STATE);
186  if(state != 0)
187  buff->addState(state);
188 
189  VectorMap<String, int>* skillMods = trapData->getSkillMods();
190  for(int i = 0; i < skillMods->size(); ++i) {
191  buff->setSkillModifier(skillMods->elementAt(i).getKey(), skillMods->get(i));
192  }
193 
194  String startSpam = trapData->getStartSpam();
195  if(!startSpam.isEmpty())
196  buff->setStartFlyText("trap/trap", startSpam, 0, 0xFF, 0);
197 
198  String stopSpam = trapData->getStopSpam();
199  if(!stopSpam.isEmpty())
200  buff->setEndFlyText("trap/trap", stopSpam, 0xFF, 0, 0);
201 
202  damage = System::random(trapData->getMaxDamage() - trapData->getMinDamage()) + trapData->getMinDamage();
203 
204  } else {
205  if(!trapData->getFailMessage().isEmpty()) {
206  message.setStringId("trap/trap" , trapData->getFailMessage());
207  }
208  }
209 
210  message.setTT(targetCreature->getDisplayedName());
211 
212 
213  Reference<ThrowTrapTask*> trapTask = new ThrowTrapTask(creature, targetCreature, buff, message, trapData->getPoolToDamage(), damage, hit);
214  creature->addPendingTask("throwtrap", trapTask, 2300);
215 
216  //Reduce cost based upon player's strength, quickness, and focus if any are over 300
217  int healthCost = creature->calculateCostAdjustment(CreatureAttribute::STRENGTH, trapData->getHealthCost());

220 
221  creature->inflictDamage(creature, CreatureAttribute::HEALTH, healthCost, false);
222  creature->inflictDamage(creature, CreatureAttribute::ACTION, actionCost, false);
223  creature->inflictDamage(creature, CreatureAttribute::MIND, mindCost, false);
224 
225  return SUCCESS;
226 
227  } catch (Exception& e) {
228 
229  }

231  return GENERALERROR;
232  }
