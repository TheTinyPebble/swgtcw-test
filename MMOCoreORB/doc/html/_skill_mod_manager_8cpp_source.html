65 void SkillModManager::init() {
66

69 
70  if (!lua->runFile("scripts/managers/skill_mod_manager.lua")) {

76  LuaObject skillLimits = lua->getGlobalObject("skillModLimits");
77 
78  if (!skillLimits.isValidTable()) {
79  error("Invalid configuration");
80  setDefaults();

83 
84  for(int i = 1; i <= skillLimits.getTableSize(); ++i) {
85  LuaObject entry = skillLimits.getObjectAt(i);
86

91  skillModMin.put(type, min);

97  skillLimits.pop();
98  delete lua;
99  lua = NULL;
100  return;

298 bool SkillModManager::compareMods(VectorMap<String, int> mods, CreatureObject* creature, uint32 type) {
299 
300  mods.setAllowOverwriteInsertPlan();
301  mods.setNullValue(0);
302 
303  Mutex* skillModMutex = creature->getSkillModMutex();
304 
305  Locker skillModLocker(skillModMutex);
306 
307  SkillModList* skillModList = creature->getSkillModList();
308 
309  if(skillModList == NULL) {
310  error("NULL SkillmodList for " + creature->getFirstName());
311  return false;
312  }
313 
314  SkillModGroup* group = skillModList->getSkillModGroup(type);
315 
316  if(group == NULL){
317  error("NULL SkillModGroup for " + creature->getFirstName());
318  return false;
319  }
320 
321  bool match = true;
322 
323  StringBuffer compare;
324  compare << " " << "SkillMod" << " " << "Player" << " " << "Computed" << endl;
325 
326  for(int i = 0; i < group->size(); ++i) {
327  String key = group->elementAt(i).getKey();
328  int value = group->get(key);
329 
330  int currentValue = mods.get(key);
331  mods.drop(key);
332 
333  compare << " " << key << " " << value << " " << currentValue << endl;
334 
335  if(value != currentValue) {
336 
337  creature->removeSkillMod(type, key, value, true);
338  creature->addSkillMod(type, key, currentValue, true);
339 
340  match = false;
341  }
342  }
343 
344  if(!mods.isEmpty()) {
345  match = false;
346  }
347 
348  if(match == false) {
349  error(compare.toString());
350 
351  if(creature->getPlayerObject() != NULL) {
352  if(creature->getPlayerObject()->getDebug()) {
353  creature->sendSystemMessage(compare.toString());
354  }
355  }
356  }
357 
358  return match;
359 }
