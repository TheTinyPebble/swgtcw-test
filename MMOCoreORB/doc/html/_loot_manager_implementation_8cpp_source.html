241  if(level > 300)
242  level = 300;

244  String directTemplateObject = templateObject->getDirectObjectTemplate();
245

247

249  error("could not create loot object: " + directTemplateObject);

253  prototype->createChildObjects();
254

256  prototype->setSerialNumber(serial);
257  prototype->setJunkDealerNeeded(templateObject->getJunkDealerTypeNeeded());
258  float fJunkValue = templateObject->getJunkMinValue()+System::random(templateObject->getJunkMaxValue()-templateObject->getJunkMinValue());
259 
260  if (level>0 && templateObject->getJunkDealerTypeNeeded()>1){
261  fJunkValue=fJunkValue + (fJunkValue * ((float)level / 100)); // This is the loot value calculation if the item has a level
262  }
263  CraftingValues craftingValues = templateObject->getCraftingValuesCopy();
264 
265  setInitialObjectStats(templateObject, &craftingValues, prototype);
266 
267  setCustomObjectName(prototype, templateObject);
268 
269  float excMod = 1.0;
270 
271  if (System::random(legendaryChance) == legendaryChance) {
272  UnicodeString objectName = prototype->getCustomObjectName();
273  uint32 bitmask = prototype->getOptionsBitmask() | OptionBitmask::YELLOW;
274 
275  if (objectName.isEmpty())
276  objectName = StringIdManager::instance()->getStringId(prototype->getObjectName()->getFullPath().hashCode());
277 
278  UnicodeString newName = objectName + " (Legendary)";
279  prototype->setCustomObjectName(newName, false);
280 
281  excMod = legendaryModifier;
282 
283  prototype->setOptionsBitmask(bitmask, false);
284  } else if (System::random(exceptionalChance) == exceptionalChance) {
285  UnicodeString objectName = prototype->getCustomObjectName();
286  uint32 bitmask = prototype->getOptionsBitmask() | OptionBitmask::YELLOW;
287 
288  if (objectName.isEmpty())
289  objectName = StringIdManager::instance()->getStringId(prototype->getObjectName()->getFullPath().hashCode());
290 
291  UnicodeString newName = objectName + " (Exceptional)";
292  prototype->setCustomObjectName(newName, false);
293 
294  excMod = exceptionalModifier;
295 
296  prototype->setOptionsBitmask(bitmask, false);
297  }
298  String subtitle;
299  bool yellow = false;
300 
301  float percentage = System::random(10000) / 10000.f; //Generate a base percentage. We will deviate slightly from this on each stat.
302 
303  for (int i = 0; i < craftingValues.getExperimentalPropertySubtitleSize(); ++i) {
304  subtitle = craftingValues.getExperimentalPropertySubtitle(i);
305 
306  if (subtitle == "hitpoints" || subtitle == "maxrange") {
307  if(!(prototype->isComponent())) {
308  continue;
309  }
310  }
311 
312  float min = craftingValues.getMinValue(subtitle);
313  float max = craftingValues.getMaxValue(subtitle);
314 
315  if (min == max)
316  continue;
317 
318  if (subtitle != "useCount" &&
319  subtitle != "quantity" &&
320  subtitle != "charges" &&
321  subtitle != "uses" &&
322  subtitle != "charge") {
323 
324  float minMod = (max >= min) ? 2000.f : -2000.f;
325  float maxMod = (max >= min) ? 500.f : -500.f;
326 
327  min = (min * level / minMod) + min;
328  max = (max * level / maxMod) + max;
329 
330  if (max >= min && max > 0) {
331  min *= excMod;
332  max *= excMod;
333  } else if (max <= min && max > 0) {
334  min /= excMod;
335  max /= excMod;
336  } else if (max <= min && max <= 0) {
337  min *= excMod;
338  max *= excMod;
339  } else {
340  min /= excMod;
341  max /= excMod;
342  }
343  } else {
344  if (excMod != 1.0) {
345  min *= yellowModifier;
346  max *= yellowModifier;
347  }
348  }
349 
350  if (excMod == 1.0 && (yellowChance == 0 || System::random(yellowChance) == 0)) {
351  if (max >= min && max > 0) {
352  min *= yellowModifier;
353  max *= yellowModifier;
354  } else if (max <= min && max > 0) {
355  min /= yellowModifier;
356  max /= yellowModifier;
357  } else if (max <= min && max <= 0) {
358  min *= yellowModifier;
359  max *= yellowModifier;
360  } else {
361  min /= yellowModifier;
362  max /= yellowModifier;
363  }
364 
365  yellow = true;
366  }
367 
368  craftingValues.setMinValue(subtitle, min);
369  craftingValues.setMaxValue(subtitle, max);
370 
371  float deviation = (((float) System::random(400)) - 200) / 1000.f; //Deviate up to 2%
372 
373  craftingValues.setCurrentPercentage(subtitle, percentage + deviation);
374  }
375 
376  if (yellow) {
377  uint32 bitmask = prototype->getOptionsBitmask() | OptionBitmask::YELLOW;
378  prototype->setOptionsBitmask(bitmask, false);
379  prototype->setJunkValue((int)(fJunkValue * 1.25));
380  }else{
381  if (excMod==1){
382  prototype->setJunkValue((int)(fJunkValue));
383  }else{
384  prototype->setJunkValue((int)(fJunkValue * (excMod/2)));
385  }
386  }
387 
388  // Use percentages to recalculate the values
389  craftingValues.recalculateValues(false);
390 
391  craftingValues.addExperimentalProperty("creatureLevel", "creatureLevel", level, level, 0, false, CraftingManager::LINEARCOMBINE);
392  craftingValues.setHidden("creatureLevel");
393 
394  //check weapons and weapon components for min damage > max damage
395  if (prototype->isComponent() || prototype->isWeaponObject()) {
396  if (craftingValues.hasProperty("mindamage") && craftingValues.hasProperty("maxdamage")) {
397  float oldMin = craftingValues.getCurrentValue("mindamage");
398  float oldMax = craftingValues.getCurrentValue("maxdamage");
399 
400  if (oldMin > oldMax) {
401  craftingValues.setCurrentValue("mindamage", oldMax);
402  craftingValues.setCurrentValue("maxdamage", oldMin);
403  }
404  }
405  }
406 
407  // Add Dots to weapon objects.
408  addStaticDots(prototype, templateObject, level);
409  addRandomDots(prototype, templateObject, level, excMod);
410 
411  setSkillMods(prototype, templateObject, level, excMod);
412 
413  setSockets(prototype, &craftingValues);
414 
415  // Update the Tano with new values
416  prototype->updateCraftingValues(&craftingValues, true);
417 
418  //add some condition damage where appropriate
419  if (!maxCondition)
420  addConditionDamage(prototype, &craftingValues);
421 
422  return prototype;

442 void LootManagerImplementation::setSkillMods(TangibleObject* object, LootItemTemplate* templateObject, int level, float excMod) {
443  if (!object->isWeaponObject() && !object->isWearableObject())
444  return;
445 
446  VectorMap<String, int>* skillMods = templateObject->getSkillMods();
447  VectorMap<String, int> additionalMods;
448 
449  bool yellow = false;
450 
451  if (System::random(skillModChance / excMod) == 0) {
452  // if it has a skillmod the name will be yellow
453  yellow = true;
454  int modCount = 1;
455  int roll = System::random(100);
456 
457  if(roll > (100 - excMod))
458  modCount += 2;
459 
460  if(roll < (5 * excMod))
461  modCount += 1;
462 
463  for(int i = 0; i < modCount; ++i) {
464  //Mods can't be lower than -1 or greater than 25
465  int max = MAX(-1, MIN(25, round(0.1f * level + 3)));
466  int min = MAX(-1, MIN(25, round(0.075f * level - 1)));
467 
468  int mod = System::random(max - min) + min;
469 
470  if(mod == 0)
471  mod = 1;
472 
473  String modName = getRandomLootableMod( object->getGameObjectType() );
474  if( !modName.isEmpty() )
475  additionalMods.put(modName, mod);
476  }
477  }
478 
479  if (yellow) {
480  uint32 bitmask = object->getOptionsBitmask() | OptionBitmask::YELLOW;
481 
482  object->setOptionsBitmask(bitmask, false);
483  }
484 
485  if (object->isWearableObject()) {
486  ManagedReference<WearableObject*> wearableObject = cast<WearableObject*>(object);
487 
488  for (int i = 0; i < additionalMods.size(); i++) {
489  wearableObject->addSkillMod(SkillModManager::WEARABLE, additionalMods.elementAt(i).getKey(), additionalMods.elementAt(i).getValue());
490  }
491 
492  for (int i = 0; i < skillMods->size(); i++) {
493  wearableObject->addSkillMod(SkillModManager::WEARABLE, skillMods->elementAt(i).getKey(), skillMods->elementAt(i).getValue());
494  }
495  } else if (object->isWeaponObject()) {
496  ManagedReference<WeaponObject*> weaponObject = cast<WeaponObject*>(object);
497 
498  for (int i = 0; i < additionalMods.size(); i++) {
499  weaponObject->addSkillMod(SkillModManager::WEARABLE, additionalMods.elementAt(i).getKey(), additionalMods.elementAt(i).getValue());
500  }
501 
502  for (int i = 0; i < skillMods->size(); i++) {
503  weaponObject->addSkillMod(SkillModManager::WEARABLE, skillMods->elementAt(i).getKey(), skillMods->elementAt(i).getValue());
504  }
505  }
