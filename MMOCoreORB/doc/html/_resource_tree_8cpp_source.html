70 bool ResourceTree::buildTreeFromClient() {
71  TemplateManager* templateManager = TemplateManager::instance();
72 
73  IffStream* iffStream = templateManager->openIffFile("datatables/resource/resource_tree.iff");
74 
75  if (iffStream == NULL) {
76  info("The Resource Tree could not be found.", true);
77  return false;
78  }
79 
80  DataTableIff dtiff;
81  dtiff.readObject(iffStream);
82 
83  Vector<String> currentClasses;
84  Vector<String> currentStfClasses;
85 
86  baseNode = new ResourceTreeNode("resource", "Resources", 0);
87 
88  String stringvalue, randomname, resourcecontainer;
89  int intvalue;
90  bool boolValue;
91 
92  for (int i = 1; i < dtiff.getTotalRows(); ++i) {
93 
94  DataTableRow* row = dtiff.getRow(i);
95 
96  row->getValue(1, stringvalue);
97 
98  ResourceTreeEntry* entry = new ResourceTreeEntry(stringvalue);
99 
100  for (int j = 3; j <= 9; ++j) {
101  String resourceclass;
102  row->getValue(j, resourceclass);
103 
104  if (resourceclass.isEmpty())
105  continue;
106 
107  while (currentStfClasses.size() > j - 3) {
108  currentStfClasses.removeElementAt(j - 3);
109  currentClasses.removeElementAt(j - 3);
110  }
111 
112  currentStfClasses.add(stringvalue);
113  currentClasses.add(resourceclass);
114  }
115 
116  for (int j = 0; j < currentStfClasses.size(); ++j) {
117  entry->addClass(currentClasses.get(j));
118  entry->addStfClass(currentStfClasses.get(j));
119  }
120 
121  row->getValue(10, intvalue);
122  entry->setMaxtype(intvalue);
123 
124  row->getValue(11, intvalue);
125  entry->setMintype(intvalue);
126 
127  row->getValue(12, intvalue);
128  entry->setMinpool(intvalue);
129 
130  row->getValue(13, intvalue);
131  entry->setMaxpool(intvalue);
132 
133  row->getValue(14, boolValue);
134  entry->setRecycled(boolValue);
135 
136  for (int j = 16; j <= 26; ++j) {
137  row->getValue(j, stringvalue);
138  if (stringvalue == "")
139  break;
140 
141  int min;
142  row->getCell(j + 11 + (j - 16))->getValue(min);
143 
144  int max;
145  row->getCell(j + 12 + (j - 16))->getValue(max);
146 
147  entry->addAttribute(new ResourceAttribute(stringvalue, min, max));
148  }
149 
150  row->getCell(49)->getValue(stringvalue);
151  if(stringvalue != "")
152  resourcecontainer = stringvalue;
153  entry->setResourceContainerType(resourcecontainer);
154 
155  row->getCell(50)->getValue(stringvalue);
156  if(stringvalue != "")
157  randomname = stringvalue;
158  entry->setRandomNameClass(randomname);
159 
160  setZoneRestriction(entry);
161  setJtl(entry);
162  setRecycled(entry);
163  setSurveyToolType(entry);
164  setRecycleToolType(entry);
165 
167  baseNode->add(entry);
168  }
170  //baseNode->updateEntries();
171 
172  return true;
173 }
