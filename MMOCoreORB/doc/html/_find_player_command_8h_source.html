63

103

106

111

119  }
120 
121  Quaternion* dir = targetObject->getDirection();
122  text << "Direction:\t\t {x:" << dir->getX() << ", y:" << dir->getY() << ", z:" << dir->getZ() << ", w:" << dir->getW() << "}\n";
123 
124  suiBox->setPromptText(text.toString());

126  player->sendMessage(suiBox->generateMessage());
