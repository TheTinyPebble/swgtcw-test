53 int SaberInventoryContainerComponent::canAddObject(SceneObject* sceneObject, SceneObject* object, int containmentType, String& errorDescription) {
54 
55  if (!object->isLightsaberCrystalObject()) {
56  errorDescription = "@jedi_spam:saber_not_crystal";
57  return TransferErrorCode::INVALIDTYPE;
58  }
59 
60 
61  LightsaberCrystalComponent* crystal = cast<LightsaberCrystalComponent*> (object);
62 
63  if (crystal->getOwner() == ""){
64  errorDescription = "@jedi_spam:saber_crystal_not_tuned";
65  return TransferErrorCode::INVALIDTYPE;
66  }
67 
68  ManagedReference<CreatureObject*> creature = cast<CreatureObject*>(object->getParent().get().get());
69 
70  if (creature != NULL && crystal->getOwner() != creature->getDisplayedName()){
71  errorDescription = "@jedi_spam:saber_crystal_not_owner";
72  return TransferErrorCode::INVALIDTYPE;
73  }
74 
75  if (containmentType == 4){
76  errorDescription = "@jedi_spam:saber_crystal_not_while_equipped";
77  return TransferErrorCode::INVALIDTYPE;
78  }
79 
80  VectorMap<uint64, ManagedReference<SceneObject*> >* containerObjects = sceneObject->getContainerObjects();
81 
82  if (containerObjects->size() >= sceneObject->getContainerVolumeLimit()) {
83  errorDescription = "@container_error_message:container03"; //This container is full.
84  return TransferErrorCode::CONTAINERFULL;
85  }
86 
87 
88  for (int i = 0; i < containerObjects->size(); i++){
89  Reference<LightsaberCrystalComponent*> crystalInside = sceneObject->getContainerObject(i).castTo<LightsaberCrystalComponent*>();
90 
91  if (crystal->getColor() != 31 && crystalInside->getColor() != 31){
92  errorDescription = "@jedi_spam:saber_already_has_color";
93  return TransferErrorCode::INVALIDTYPE;
94  }
95  }
96 
97  return 0;
98 }

104 int SaberInventoryContainerComponent::notifyObjectInserted(SceneObject* sceneObject, SceneObject* object) {
105  ManagedReference<WeaponObject*> weao = cast<WeaponObject*>( sceneObject->getParent().get().get());
106 
107  if (weao->isJediWeapon()) {
108  ManagedReference<LightsaberCrystalComponent*> crystal = cast<LightsaberCrystalComponent*>( object);
109  if (crystal->getColor() == 31){
110  weao->setAttackSpeed(weao->getAttackSpeed() + crystal->getAttackSpeed());
111  weao->setMinDamage(weao->getMinDamage() + crystal->getMinimumDamage());
112  weao->setMaxDamage(weao->getMaxDamage() + crystal->getMaximumDamage());
113  weao->setHealthAttackCost(weao->getHealthAttackCost() + crystal->getSacHealth());
114  weao->setActionAttackCost(weao->getActionAttackCost() + crystal->getSacAction());
115  weao->setMindAttackCost(weao->getMindAttackCost() + crystal->getSacMind());
116  weao->setWoundsRatio(weao->getWoundsRatio() + crystal->getWoundChance());
117  weao->setForceCost(weao->getForceCost() + crystal->getForceCost());
118  }
119  if (crystal->getColor() != 31) {
120  int color = crystal->getColor();
121  weao->setBladeColor(color);
122  weao->setCustomizationVariable("/private/index_color_blade", color, true);
123  }
124  }
125 
126  return sceneObject->notifyObjectInserted(object);
127 }
