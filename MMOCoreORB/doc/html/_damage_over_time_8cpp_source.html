133 uint32 DamageOverTime::applyDot(CreatureObject* victim) {

136  switch(type) {

139  return 0;

148  case CreatureState::POISONED:

151  }
152  else if (nextTick.isPast()) {
153  power = doPoisonTick(victim);
154 
155  nextTick.updateToCurrentTime();
156  nextTick.addMiliTime(10000);
157  }
158  break;
159  case CreatureState::DISEASED:
160  if (expires.isPast()) {
161  return 0;
162  }
163  else if (nextTick.isPast()) {
164  power = doDiseaseTick(victim);
165 
166  nextTick.updateToCurrentTime();
167  nextTick.addMiliTime(40000);
168  }
169  break;
170  case CreatureState::ONFIRE:
171  if (expires.isPast()) {
172  return 0;
173  }
174  else if (nextTick.isPast()) {
175  power = doFireTick(victim);
176 
177  nextTick.updateToCurrentTime();
178  nextTick.addMiliTime(10000);
179  }

181  }

287 float DamageOverTime::reduceTick(float reduction) {
288  //System::out << "reducing tick with reduction " << reduction << endl;
289  if (reduction < 0)
290  return reduction;
291 
292  float effReduction = (strength - (float)reduction);
293  float reductionLeft = (float)reduction - strength;
294 
295  if (reductionLeft >= 0.0f) {
296  expires.updateToCurrentTime();
297  } else {
298  //System::out << "strength before dotRed " << strength << endl;
299  strength = effReduction;
300  //System::out << "strength after dotRed " << strength << endl;
301  }
302 
303  return reductionLeft;
304 }
