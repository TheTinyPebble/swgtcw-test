166 void ResourceSpawner::loadResourceSpawns() {
167 
168  ObjectDatabase* resourceDatabase =
169  ObjectDatabaseManager::instance()->loadObjectDatabase("resourcespawns",
170  true);
171 
172  ObjectDatabaseManager::instance()->commitLocalTransaction();
173  ObjectDatabaseIterator iterator(resourceDatabase, true);
174 
175  uint64 objectID = 0;
176 
177  while (iterator.getNextKey(objectID)) {
178 
179  Reference<ResourceSpawn*> resourceSpawn = Core::getObjectBroker()->lookUp(objectID).castTo<ResourceSpawn*>();
180  //ObjectDatabaseManager::instance()->commitLocalTransaction();
181 
182  if (resourceSpawn == NULL) {
183  error("Trying to load object as ResourceSpawn that is not a resource spawn");
184  continue;
185  }
186 
187  resourceMap->add(resourceSpawn->getName(), resourceSpawn);
188 
189  if (!resourceSpawn->inShift()) {
190  despawn(resourceSpawn);
191  continue;
192  }
193 
194  if (resourceSpawn->getSpawnPool() != 0) {
195 
196  switch (resourceSpawn->getSpawnPool()) {
197  case ResourcePool::MINIMUMPOOL:
198  minimumPool->addResource(resourceSpawn, resourceSpawn->getPoolSlot());
199  break;
200  case ResourcePool::RANDOMPOOL:
201  randomPool->addResource(resourceSpawn, resourceSpawn->getPoolSlot());
202  break;
203  case ResourcePool::FIXEDPOOL:
204  fixedPool->addResource(resourceSpawn, resourceSpawn->getPoolSlot());
205  break;
206  case ResourcePool::NATIVEPOOL:
207  nativePool->addResource(resourceSpawn, resourceSpawn->getPoolSlot());
208  break;
209  case ResourcePool::MANUALPOOL:
210  manualPool->addResource(resourceSpawn, resourceSpawn->getPoolSlot());
211  break;
212  }
213  }
214  }
215 
216  if(resourceMap->size() == 0 && scriptLoading) {
217 
218  spawnScriptResources();
219 
220  }
221 
222  String built = "Resource Map Built with " + String::valueOf(
223  resourceMap->size()) + " resources";
224  info(built, true);
225 
226 }

228 void ResourceSpawner::spawnScriptResources() {
229 
230  Lua* lua = new Lua();
231  lua->init();
232 
233  if (!lua->runFile("scripts/managers/resource_manager_spawns.lua")) {
234  error("Invalid script = 'scripts/managers/resource_manager_spawns.lua'");
235  delete lua;
236  return;
237  }
238 
239  LuaObject luaObject = lua->getGlobalObject("resources");
240  if (!luaObject.isValidTable()) {
241  error("Invalid table = 'scripts/managers/resource_manager_spawns.lua'");
242  delete lua;
243  return;
244  }
245 
246  for(int i = 1; i <= luaObject.getTableSize(); ++i) {
247 
248  LuaObject resource = luaObject.getObjectAt(i);
249  if(!resource.isValidTable())
250  continue;
251 
252  Reference<ResourceSpawn*> newSpawn = dynamic_cast<ResourceSpawn*>
253  (objectManager->createObject(0xb2825c5a, 1, "resourcespawns"));
254 
255  if (newSpawn == NULL) {
256  error("createResourceSpawn is trying to create a resourcespawn with the wrong type");
257  continue;
258  }
259 
260  newSpawn->setType(resource.getStringField("type"));
261  newSpawn->setName(resource.getStringField("name"));
262 
263  LuaObject classes = resource.getObjectField("classes");
264 
265  for (int j = 1; j <= classes.getTableSize(); ++j) {
266  LuaObject classesEntry = classes.getObjectAt(j);
267 
268  newSpawn->addClass(classesEntry.getStringAt(1));
269  newSpawn->addStfClass(classesEntry.getStringAt(2));
270 
271  classesEntry.pop();
272  }
273 
274  resource.pop();
275 
276  LuaObject attributes = resource.getObjectField("attributes");
277 
278  for (int j = 1; j <= attributes.getTableSize(); ++j) {
279  LuaObject attributesEntry = attributes.getObjectAt(j);
280 
281  newSpawn->addAttribute(attributesEntry.getStringAt(1), attributesEntry.getIntAt(2));
282 
283  attributesEntry.pop();
284  }
285 
286  resource.pop();
287 
288  newSpawn->setDespawned(time(0));
289 
290  newSpawn->setZoneRestriction(resource.getStringField("zoneRestriction"));
291  newSpawn->setSurveyToolType(resource.getIntField("surveyToolType"));
292  newSpawn->setContainerCRC(resource.getIntField("containerCRC"));
293 
294  if (newSpawn->isType("energy") || newSpawn->isType("radioactive"))
295  newSpawn->setIsEnergy(true);
296 
297  resourceMap->add(newSpawn->getName(), newSpawn);
298 
299  luaObject.pop();
300  }

303 }

551 int ResourceSpawner::randomizeValue(int min, int max) {
552  if (min == 0 && max == 0)
553  return 0;
554 
555  if (min > lowerGateOverride)
556  min = lowerGateOverride;
557 
558  int breakpoint = (int) (spawnThrottling * (max - min)) + min;
559  int randomStat = System::random(max - min) + min;
560  bool aboveBreakpoint = System::random(10) == 7;
561 
562  if (!(aboveBreakpoint && randomStat > breakpoint) || (!aboveBreakpoint
563  && randomStat < breakpoint)) {
564 
565  if (aboveBreakpoint) {
566  while (randomStat < breakpoint)
567  randomStat = System::random(max - min) + min;
568  } else {
569  while (randomStat > breakpoint)
570  randomStat = System::random(max - min) + min;
571  }
572  }
573  return randomStat;
574 }
