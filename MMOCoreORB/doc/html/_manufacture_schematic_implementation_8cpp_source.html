302 void ManufactureSchematicImplementation::initializeIngredientSlots() {
303 
304  Locker locker(_this.get());
305 
306  if(draftSchematic == NULL || initialized)
307  return;
308 
309  ingredientSlots.removeAll();
310  ingredientNames.removeAll(NULL);
311  ingredientTypes.removeAll(NULL);
312  slotOIDs.removeAll(NULL);
313  slotQuantities.removeAll(NULL);
314  slotQualities.removeAll(NULL);
315  slotClean.removeAll(NULL);;
316  slotIndexes.removeAll(NULL);
317  craftingValues->clearAll();
318 
319  assembled = false;
320  completed = false;
321  initialized = true;
322  complexity = draftSchematic->getComplexity();
323  manufactureLimit = 0;
324  ingredientCounter = draftSchematic->getDraftSlotCount() * 4;
325 
326  for (int i = 0; i < draftSchematic->getDraftSlotCount(); ++i) {
327 
328  Reference<IngredientSlot* > ingredientSlot = NULL;
329  Reference<DraftSlot* > draftSlot = draftSchematic->getDraftSlot(i);
330 
331  ingredientNames.add(StringId(draftSlot->getStringId()));
332  ingredientTypes.add(0);
333  slotOIDs.add(Vector<uint64>());
334  slotQuantities.add(Vector<int>());
335  slotQualities.add(0);
336  slotClean.add(0xFFFFFFFF);
337  slotIndexes.add(i);
338 
339  switch (draftSlot->getSlotType()) {
340  case IngredientSlot::RESOURCESLOT:
341  ingredientSlot = new ResourceSlot();
342  ingredientSlot->setOptional(false);
343  ingredientSlot->setIdentical(true);
344  break;
345  case IngredientSlot::IDENTICALSLOT:
346  ingredientSlot = new ComponentSlot();
347  ingredientSlot->setOptional(false);
348  ingredientSlot->setIdentical(true);
349  break;
350  case IngredientSlot::MIXEDSLOT:
351  ingredientSlot = new ComponentSlot();
352  ingredientSlot->setOptional(false);
353  ingredientSlot->setIdentical(false);
354  break;
355  case IngredientSlot::OPTIONALIDENTICALSLOT:
356  ingredientSlot = new ComponentSlot();
357  ingredientSlot->setOptional(true);
358  ingredientSlot->setIdentical(true);
359  break;
360  case IngredientSlot::OPTIONALMIXEDSLOT:
361  ingredientSlot = new ComponentSlot();
362  ingredientSlot->setOptional(true);
363  ingredientSlot->setIdentical(false);
364  break;
365  default:
366  error("Ingredient Slot script value bad: " + draftSchematic->getDisplayedName());
367  continue;
368  }
369 
370  ingredientSlot->setContentType(draftSlot->getResourceType());
371  ingredientSlot->setQuantityNeeded(draftSlot->getQuantity());
372  ingredientSlot->setSlotName(draftSlot->getStringId().getStringID());
373 
374  ingredientSlots.add(ingredientSlot.get());
375  }

396  wasEmpty = true;
397 
398  if(!ingredientSlot->add(player, satchel, tano))
399  return IngredientSlot::INVALIDINGREDIENT;
400 
401 
402  if(wasEmpty) {
403

405  increaseComplexity();
406 
407  // DMSCO6 ***************************************************

535 bool ManufactureSchematicImplementation::isReadyForAssembly() {
536 
537  if(ingredientSlots.isEmpty() || !initialized)
538  return false;
539  // store off all component objects we find
540  int componetSlotItemCount = 0;
541  HashSet<uint64> usedObjectIds;
542  bool usingIdentical = false;
543  for (int i = 0; i < ingredientSlots.size(); ++i) {
544 
545  Reference<IngredientSlot* > slot = ingredientSlots.get(i);
546 
547  // null slots should happen unless something bad happened
548  if(slot == NULL)
549  return false;
550 
551  if(slot->isOptional()) // Skip before adding as it can be blank
552  continue;
553 
554  if(!slot->isFull())
555  return false;
556 
557  if (slot->isComponentSlot()) {
558  Vector<uint64> v = slot->getOIDVector();
559  componetSlotItemCount += slot->getSlotQuantity();
560  for (int i=0;i<v.size();i++) {
561  usedObjectIds.add(v.get(i));
562  }
563  }
564  // If we are using idenitcal slots, we may have duplicate objectid i.e. factory crate
565  if(slot->requiresIdentical())
566  usingIdentical = true;
567  }
568  if (componetSlotItemCount != usedObjectIds.size())
569  if(!usingIdentical)
570  return false;
571  return true;
572 }
