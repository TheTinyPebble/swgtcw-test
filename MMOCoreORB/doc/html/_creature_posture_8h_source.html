203  void loadMovementData() {

205 
206  if (iffStream == NULL) {
207  error("Could not load movement data.");

210 
211  DataTableIff dtiff;
212  dtiff.readObject(iffStream);
213 
214  delete iffStream;
215 
216  for (int i = 0; i < dtiff.getTotalRows(); i++) {
217  DataTableRow* row = dtiff.getRow(i);
218  CreatureMovementEntry entry;
219  int value;
220  float scale;
221 
222  row->getValue(0, value);
223  if (value < 0) entry.posture = 0xFF;
224  else entry.posture = (uint8)value;
225 
226  row->getValue(1, value);

228  else entry.stationary = (uint8)value;

233 
234  row->getValue(3, value);
