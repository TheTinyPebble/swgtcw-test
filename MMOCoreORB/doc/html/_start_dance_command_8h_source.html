62  static void startDance(CreatureObject* creature, const String& dance,
63  const String& animation) {

65  SessionFacadeType::ENTERTAINING);
66  ManagedReference<EntertainingSession*> session =

143 
144  String args = arguments.toString();
145 
146  PerformanceManager* performanceManager =

148 
149  if (args.length() < 2) {

151  return SUCCESS;

153 
154  String fullString = String("startDance") + "+" + args;
155 
156  if (!ghost->hasAbility(fullString)) {
157  creature->sendSystemMessage("@performance:dance_lack_skill_self");
158  return GENERALERROR;

162  creature->sendSystemMessage("@performance:dance_lack_skill_self");
163  return GENERALERROR;
