534 void SkillManager::updateXpLimits(PlayerObject* ghost) {

537  }
538 
539  VectorMap<String, int>* xpTypeCapList = ghost->getXpTypeCapList();
540 
541  //Clear all xp limits to the default limits.
542  for (int i = 0; i < defaultXpLimits.size(); ++i) {
543  String xpType = defaultXpLimits.elementAt(i).getKey();
544  int xpLimit = defaultXpLimits.elementAt(i).getValue();
545 
546  if (xpTypeCapList->contains(xpType)) {
547  xpTypeCapList->get(xpType) = xpLimit;
548  } else {
549  xpTypeCapList->put(xpType, xpLimit);
550  }
551  }
552 
553  //Iterate over the player skills and update xp limits accordingly.
554  ManagedReference<CreatureObject*> player = cast<CreatureObject*>(ghost->getParentRecursively(SceneObjectType::PLAYERCREATURE).get().get());
555 
556  if(player == NULL)
557  return;
558 
559  SkillList* playerSkillBoxList = player->getSkillList();
560 
561  for(int i = 0; i < playerSkillBoxList->size(); ++i) {
562  Skill* skillBox = playerSkillBoxList->get(i);
563  if (xpTypeCapList->contains(skillBox->getXpType()) && (xpTypeCapList->get(skillBox->getXpType()) < skillBox->getXpCap())) {
564  xpTypeCapList->get(skillBox->getXpType()) = skillBox->getXpCap();
565  }
566  }
567 
568  //Iterate over the player xp types and cap all xp types to the limits.
569  DeltaVectorMap<String, int>* experienceList = ghost->getExperienceList();
570 
571  for (int i = 0; i < experienceList->size(); ++i) {
572  String xpType = experienceList->getKeyAt(i);
573  if (experienceList->get(xpType) > xpTypeCapList->get(xpType)) {
574  ghost->addExperience(xpType, xpTypeCapList->get(xpType) - experienceList->get(xpType), true);
575  }
576  }
