84  void insertToMessage(BaseMessage* msg) {
85  msg->insertAscii(stringId.getFile()); // ex: craft_food_ingredients_n
86  msg->insertInt(0);
87  msg->insertAscii(stringId.getStringID()); // ex: dried_fruit
88 
89  if (slotType == OPTIONALIDENTICALSLOT || slotType == OPTIONALMIXEDSLOT)
90  msg->insertByte(1); // ex: additive is optional so insertByte(1);
91  else
92  msg->insertByte(0);
93 
94  msg->insertInt(1);
95 
96  msg->insertAscii(stringId.getFile()); // ex: craft_food_ingredients_n
97  msg->insertInt(0);
98  msg->insertAscii(stringId.getStringID()); // ex: dried_fruit
99 
100  UnicodeString uniResourceType(resourceType);
101  msg->insertUnicode(uniResourceType); // ex: organic
102 
103  if (slotType == MIXEDSLOT || slotType == OPTIONALMIXEDSLOT) {
104  msg->insertByte(5); // Enables Components
105  msg->insertInt(quantity); // ex: 3
106  } else if (slotType == IDENTICALSLOT || slotType == OPTIONALIDENTICALSLOT) {
107  msg->insertByte(2);
108  msg->insertInt(quantity); // ex: 3
109  msg->insertShort(0);
110  } else if (slotType == RESOURCESLOT) {
111  msg->insertByte(4);
112  msg->insertInt(quantity); // ex: 3
113  }
114  }
