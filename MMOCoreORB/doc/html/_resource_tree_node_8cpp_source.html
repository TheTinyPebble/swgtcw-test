90 void ResourceTreeNode::add(ResourceTreeEntry* entry) {
91 
92 
93  //Find out which child node this entry belongs to, and
94  //if it doesn't exist, create it
95  for(int ii = 0; ii < nodes.size(); ++ii) {
96 
97  ResourceTreeNode* node = nodes.get(ii);
98  if(node->getName() == entry->getClass(depth)) {
99  node->add(entry);
100  return;
101  }
102  }
103 
104  //The entry doesn't belong to an existing child node, now we
105  //determine if it belong on this node, or on a new child node.
106  if(entry->getClassCount() > depth + 1) {
107 
108  String stfType = "";
109 
110  for(int i = 0; i < entries.size(); ++i) {
111  ResourceTreeEntry* ent = entries.get(i);
112  if(ent->getFinalClass() == entry->getClass(depth)) {
113  stfType = ent->getType();
114  ent->setChildren(true);
115  break;
116  }
117  }
118 
119  ResourceTreeNode* newnode = new ResourceTreeNode(stfType, entry->getClass(depth), depth + 1);
120  nodes.add(newnode);
121  newnode->add(entry);
122  newnode->setParentNode(this);
123 
124  } else {
125 
126  entry->setMyNode(this);
127  entries.add(entry);
128  }

166 ResourceTreeEntry* ResourceTreeNode::getEntry(const String& type,
167  const Vector<String>& excludes, const String& zoneName) {
168 
169  ResourceTreeEntry* entry = NULL;
170  entry = find(type, entry);
171 
172  if(entry == NULL)
173  return NULL;
174 
175  // If the entry has no childen, it means
176  // that a specific resource was requested
177  if(!entry->hasChildren())
178  return entry;
179 
180  ResourceTreeNode* node = entry->getMyNode()->findNode(type);
181 
182  if(node == NULL)
183  return NULL;
184 
185  if (zoneName != "") {
186 
187  return node->getPlanetSpecificEntry(zoneName);
188 
189  } else {
190 
191  Vector<ResourceTreeEntry*> candidates;
192 
193  node->getEntryPool(candidates, excludes);
194 
195  if(candidates.size() == 0)
196  return NULL;
197 
198  int random = System::random(candidates.size() - 1);
199 
200  return candidates.get(random);

214 void ResourceTreeNode::getEntryPool(Vector<ResourceTreeEntry*>& candidates,
215  const Vector<String> excludes) {
216  for(int i = 0; i < entries.size(); ++i) {
217 
218  ResourceTreeEntry* ent = entries.get(i);
219  bool valid = true;
220 
221  if(!ent->isRecycled() && !ent->hasChildren()) {
222

224  if(ent->isType(excludes.get(ii))) {
225  valid = false;
