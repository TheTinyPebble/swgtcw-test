56 int ConversationObserverImplementation::notifyObserverEvent(unsigned int eventType, Observable* observable, ManagedObject* arg1, long long arg2) {
57  //Verify needed parameters
58  if (eventType != ObserverEventType::CONVERSE && eventType != ObserverEventType::STARTCONVERSATION &&
59  eventType != ObserverEventType::SELECTCONVERSATION && eventType != ObserverEventType::STOPCONVERSATION &&
60  eventType != ObserverEventType::POSITIONCHANGED) {
61  //Incorrect event type.
62  return 0;
63  }
64 
65  if (observable == NULL)
66  return 0;
67 
68  if (arg1 == NULL && eventType != ObserverEventType::POSITIONCHANGED)
69  return 0;
70 
71  //Try to convert parameters to correct types.
72  CreatureObject* npc = NULL;
73  CreatureObject* player = NULL;
74  int selectedOption;
75  try {
76  npc = cast<CreatureObject* >(observable);
77 
78  if (arg1 != NULL)
79  player = cast<CreatureObject* >(arg1);
80 
81  if (arg2 < std::numeric_limits<int>::min()) {
82  selectedOption = std::numeric_limits<int>::min();
83  } else if (arg2 > std::numeric_limits<int>::max()) {
84  selectedOption = std::numeric_limits<int>::max();
85  } else {
86  selectedOption = arg2;
87  }
88  } catch (...) {
89  //Failed to convert parameters. Keep observer.
90  return 0;
91  }
92 
93  switch (eventType) {
94  case ObserverEventType::POSITIONCHANGED:
95  if (npc != NULL) { //the observable in this case is the player
96  ManagedReference<ConversationSession*> session = npc->getActiveSession(SessionFacadeType::CONVERSATION).castTo<ConversationSession*>();
97 
98  if (session != NULL) {
99  ManagedReference<CreatureObject*> sessionNpc = session->getNPC();
100 
101  if (sessionNpc == NULL || npc->getDistanceTo(sessionNpc) > 7.f) {
102  cancelConversationSession(npc, session->getNPC().get(), true);
103  return 0;
104  }
105  }
106 
107  }
108 
109  return 0;
110 
111  case ObserverEventType::STOPCONVERSATION:
112  cancelConversationSession(player, npc);
113  //Keep observer.
114  return 0;
115 
116  case ObserverEventType::STARTCONVERSATION: {
117  //Cancel any existing sessions.
118  cancelConversationSession(player, npc);
119  //Create a new session.
120  createConversationSession(player, npc);
121  createPositionObserver(player);
122  break;
123  }
124  default:
125  break;
126  }
127 
128  //Select next conversation screen.
129  Reference<ConversationScreen*> conversationScreen = getNextConversationScreen(player, selectedOption, npc);
130 
131  if (conversationScreen != NULL) {
132  //Modify the conversation screen.
133  conversationScreen = runScreenHandlers(player, npc, selectedOption, conversationScreen);
134  }
135 
136  //Send the conversation screen to the player.
137  sendConversationScreenToPlayer(player, npc, conversationScreen);
138 
139  if (conversationScreen == NULL)
140  cancelConversationSession(player, npc);
141 
142  //Keep the observer.
143  return 0;
144 }

155  ManagedReference<Facade*> session = conversingPlayer->getActiveSession(SessionFacadeType::CONVERSATION);

195 ConversationScreen* ConversationObserverImplementation::runScreenHandlers(CreatureObject* conversingPlayer, CreatureObject* conversingNPC, int selectedOption, ConversationScreen* conversationScreen) {
196  ConversationScreen* modifiedScreen = NULL;
197 
198  //Loop until a screen has been modified.
199  while (modifiedScreen == NULL) {
200  //Call appropriate screen handler.
201  if (screenHandlers.contains(conversationScreen->getScreenID())) {
202  //Copy the conversation screen to preserve the template.
203  modifiedScreen = screenHandlers.get(conversationScreen->getScreenID())->handleScreen(conversingPlayer, conversingNPC, selectedOption, conversationScreen->cloneScreen());
204 
205  //Check if another screen should be handled
206  if (modifiedScreen == NULL) {
207  //Get next screen.
208  String screenId = screenHandlers.get(conversationScreen->getScreenID())->getNextScreenId();
209  conversationScreen = getConversationScreen(screenId);
210  if (conversationScreen == NULL) {
211  //No new screen found.
212  return NULL;
213  }
214  }
215  } else {
216  //No screen handler found for this screen, exit loop and return it unmodified.
217  modifiedScreen = conversationScreen;
218  }
219  }
220 
221  return modifiedScreen;
222 }
