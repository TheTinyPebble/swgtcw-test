184 void ConfigManager::loadMOTD() {
185  File* file;

187 
188  try {
189  file = new File("conf/motd.txt");
190  reader = new FileReader(file);
191 
192  String line;
193 
194  while(reader->readLine(line)) {

196  }
197 
198  reader->close();

202  }
