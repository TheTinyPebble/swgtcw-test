58  int doQueueCommand(CreatureObject* creature, const uint64& target, const UnicodeString& arguments) {
59 
60  if (!checkStateMask(creature))
61  return INVALIDSTATE;
62 
63  if (!checkInvalidLocomotions(creature))
64  return INVALIDLOCOMOTION;
65 
66  if ( creature != NULL )
67  {
68  uint64 objectID = 0;
69  UnicodeTokenizer tokenizer(arguments);
70  tokenizer.setDelimeter(" ");
71 
72  // if we have an argument passed, use it
73  if (tokenizer.hasMoreTokens())
74  { try {
75  objectID = tokenizer.getLongToken();
76  } catch ( Exception& err ) {
77  creature->sendSystemMessage("INVALID OBJECT. Please specify a valid object name or objectid");
78  return INVALIDPARAMETERS;
79 
80  }
81  } else {
82  objectID = target;
83  }
84 
85  if ( objectID == 0 ){
86  creature->sendSystemMessage("You need to target an object or specify an object id: /getobjvars <objectID> ");
87  }
88 
89  ManagedReference<SceneObject*> object = server->getZoneServer()->getObject(objectID, false);
90 
91  if ( object == NULL){
92  creature->sendSystemMessage("ERROR GETTIGN OBJECT - NULL" + String::valueOf(target));
93  } else {
94 
95  String strClassName = object->getObjectNameStringIdName();
96  String strDescription = object->getDetailedDescription();
97  bool bMarkedForDelete = object->_isMarkedForDeletion();
98  bool bIsUpdated = object->_isUpdated();
99  int rCount = object.get()->getReferenceCount();
100  uint64 parentID = object->getParentID();
101  StringBuffer msg;
102  msg << endl << "OBJECTID: " << String::valueOf(objectID) << endl;
103  msg << "OBJECTTYPE: " << String::valueOf(object->getGameObjectType()) << endl;
104 
105  if(object->isCreatureObject()){
106  msg << "Creature First Name: " << object.castTo<CreatureObject*>()->getFirstName() << endl;
107  }
108 
109  msg << "CLASS: " << strClassName << endl;
110  msg << "Marked for deletion: " << String::valueOf(bMarkedForDelete) << endl;
111  msg << "IsUpdated: " << String::valueOf(bIsUpdated) << endl;
112  msg << "REFERENCE COUNT " << String::valueOf(rCount) << endl;
113  msg << "Path: " << object->getObjectTemplate()->getFullTemplateString() << endl;
114  msg << "Children: " << String::valueOf(object->getChildObjects()->size()) << endl;
115  msg << "PARENT OBJECTID: " << String::valueOf(parentID) << endl;
116  if(object->getZone() != NULL)
117  msg << "location: " << String::valueOf(object->getPositionX()) << " " << String::valueOf(object->getPositionY()) << " " << object->getZone()->getZoneName();
118 
119  creature->sendSystemMessage(msg.toString());
120  }
121  }
122  return SUCCESS;
123 
124  }
