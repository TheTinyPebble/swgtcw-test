90  bool fulfillsRequirements(NpcSpawnPoint* npc, const Vector3* position, const float minDistance, const float maxDistance, int spawnType) {
91  if (npc != NULL) {
92  if (((npc->getSpawnType() & spawnType) == spawnType)) {
93  float squaredDistance = npc->getPosition()->squaredDistanceTo(*position);
94  if ((squaredDistance <= maxDistance * maxDistance) &&
95  (squaredDistance >= minDistance * minDistance)) {
96  //NPC matches requirements.
97  return true;

99  }
100  }

134  NpcSpawnPoint* getRandomNpcSpawnPoint(const Vector3* position, const int spawnType, const float minDistance = 0.0, const float maxDistance = 100000.0) {
135  if (npcSpawnMap.size() == 0) {
136  return NULL;
137  }
138 
139  //Try 100 random npc spawn points, return the first that fulfills the requirements.
140  int maximumNumberOfTries = (npcSpawnMap.size() / 4) + 1;
141  while (maximumNumberOfTries > 0) {
142  int npcNumber = System::random(npcSpawnMap.size() - 1);
143 
144  NpcSpawnPoint* npc = npcSpawnMap.get(npcNumber);
145 
146  if (fulfillsRequirements(npc, position, minDistance, maxDistance, spawnType)) {
147  return npc;
148  }
149 
150  maximumNumberOfTries--;
151  }
152 
153  //100 random npc spawn points failed, do a full iteration and pick the first npc that match.
154  for (int i = 0; i < npcSpawnMap.size(); ++i) {
155  NpcSpawnPoint* npc = npcSpawnMap.get(i);
156 
157  if (fulfillsRequirements(npc, position, minDistance, maxDistance, spawnType)) {
158  return npc;
159  }
160  }
161 
162  //No npc matches the requirements.
163  return NULL;
164  }

179  NpcSpawnPoint* getNearestNpcSpawnPoint(Vector3* position) {
180  float minimumSquaredDistance = 100000.0f * 100000.0f;
181  NpcSpawnPoint* nearestNpcSpawnPoint = NULL;
182  //Iterate over all spawn points.
183  for (int i = 0; i < npcSpawnMap.size(); i++) {
184  //Calculate distance between spawn point and supplied position and store the smallest distance.
185  float squaredDistance = npcSpawnMap.get(i)->getPosition()->squaredDistanceTo(*position);
186  if (minimumSquaredDistance > squaredDistance) {
187  minimumSquaredDistance = squaredDistance;
188  nearestNpcSpawnPoint = npcSpawnMap.get(i);
189  }
190  }
191 
192  return nearestNpcSpawnPoint;
