134 void ResourceContainerImplementation::split(int newStackSize) {
135  if (getQuantity() <= newStackSize)
136  return;
137 
138  if(newStackSize > getQuantity())
139  newStackSize = getQuantity();
140 
141  ManagedReference<ResourceContainer*> newResource = spawnObject->createResource(newStackSize);
142 
143  if(parent == NULL || newResource == NULL || newResource->getSpawnObject() == NULL)
144  return;
145 
146  ManagedReference<SceneObject*> sceneParent = cast<SceneObject*>(parent.get().get());
147 
148  if(sceneParent->transferObject(newResource, -1, true)) {
149  sceneParent->broadcastObject(newResource, true);
150 
151  setQuantity(getQuantity() - newStackSize);
152  } else {
153  StringBuffer errorMessage;
154  errorMessage << "Unable to split resource in container type: " << sceneParent->getGameObjectType() << " " << sceneParent->getDisplayedName();
155  error(errorMessage.toString());
156  }
157 }
