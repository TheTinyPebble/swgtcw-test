131  bool returnToParents(CreatureObject* player) {
132 
133  if(parents.size() == 0)
134  return true;
135 
136  if(currentSpawn == NULL) {
137  warning("Spawn is null when trying to return resources");
138  return false;
139  }
140 
141  for(int i = 0; i < parents.size(); ++i) {
142 
143  SceneObject* parent = parents.elementAt(i).getKey();
144 
145  if(parent == NULL)
146  continue;
147 
148  bool found = false;
149  // Check inventory for resource and add if existing
150  for (int j = 0; j < parent->getContainerObjectsSize(); ++j) {
151  ManagedReference<SceneObject*> object = parent->getContainerObject(j);
152 
153  if (object->isResourceContainer()) {
154  ManagedReference<ResourceContainer*> resource = cast<ResourceContainer*>( object.get());
155 
156  if (resource->getSpawnName() == currentSpawn->getName()) {
157  resource->setQuantity(resource->getQuantity() + parents.get(i));
158  found = true;
159  break;
160  }
161  }
162  }
163 
164  if(!found) {
165  ManagedReference<ResourceContainer*> newContainer = currentSpawn->createResource(parents.get(i));
166  if(newContainer != NULL && newContainer->getQuantity() > 0) {
167  if(parent->transferObject(newContainer, -1, false)) {
168  parent->broadcastObject(newContainer, true);
169  } else {
170  error("Unable to return resource to parent, transfer failed");
171  }
172  } else {
173  error("Unable to return resource to parent, NULL container");
174  }
175 
176  }
177  }
178 
179  parents.removeAll();
180  currentSpawn = NULL;
181  quantity = 0;
182  return true;
183  }
