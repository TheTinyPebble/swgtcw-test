58 int LairObserverImplementation::notifyObserverEvent(unsigned int eventType, Observable* observable, ManagedObject* arg1, int64 arg2) {
59  switch (eventType) {
60  case ObserverEventType::OBJECTDESTRUCTION:
61  notifyDestruction(cast<TangibleObject*>(observable), cast<TangibleObject*>(arg1), (int)arg2);
62  return 1;
63  break;
64  case ObserverEventType::DAMAGERECEIVED:
65 
66  int livingCreatureCount = getLivingCreatureCount();
67 
68  // if there are living creatures, make them aggro
69  if(livingCreatureCount > 0 ){
70  Reference<LairAggroTask*> task = new LairAggroTask(cast<TangibleObject*>(observable), cast<TangibleObject*>(arg1), _this.get());
71  task->execute();
72  }
73 
74  // if new creatures have spawned or there are live creatures near the lair
75  if( checkForNewSpawns(cast<TangibleObject*>(observable)) || livingCreatureCount > 0 )
76  checkForHeal(cast<TangibleObject*>(observable), cast<TangibleObject*>(arg1));
77 
78  break;
79  }
80 
81  return 0;
82 }

195 
196  if (spawnedCreatures.size() >= lairTemplate->getSpawnLimit())

202  return false;

210  break;

217  break;

224  break;
225  case 3:
226  return false;

244

251

253

261

277  }
278  }
279 
280  return amountToSpawn > 0;
281 }
