75 int CraftingManagerImplementation::calculateAssemblySuccess(CreatureObject* player,
76  DraftSchematic* draftSchematic, float effectiveness) {
77 
78  // assemblyPoints is 0-12
80  float cityBonus = player->getSkillMod("private_spec_assembly");
81 
82  float assemblyPoints = ((float)player->getSkillMod(draftSchematic->getAssemblySkill())) / 10.0f;
83  int failMitigate = (player->getSkillMod(draftSchematic->getAssemblySkill()) - 100 + cityBonus) / 7;
84 
85  if(failMitigate < 0)
86  failMitigate = 0;
87  if(failMitigate > 5)
88  failMitigate = 5;
89 
90  // 0.85-1.15
91  float toolModifier = 1.0f + (effectiveness / 100.0f);
92 
93  //Pyollian Cake
94 
95  float craftbonus = 0;
96  if (player->hasBuff(BuffCRC::FOOD_CRAFT_BONUS)) {
97  Buff* buff = player->getBuff(BuffCRC::FOOD_CRAFT_BONUS);
98 
99  if (buff != NULL) {
100  craftbonus = buff->getSkillModifierValue("craft_bonus");
101  toolModifier *= 1.0f + (craftbonus / 100.0f);
102  }
103  }
104 
105  int luckRoll = System::random(100) + cityBonus;
106 
107  if(luckRoll > (95 - craftbonus))
108  return AMAZINGSUCCESS;
109 
110  if(luckRoll < (5 - craftbonus - failMitigate))
111  luckRoll -= System::random(100);
112 
113  //if(luckRoll < 5)
114  // return CRITICALFAILURE;
115 
116  luckRoll += System::random(player->getSkillMod("luck") + player->getSkillMod("force_luck"));
117 
118  int assemblyRoll = (toolModifier * (luckRoll + (assemblyPoints * 5)));
119 
120  if (assemblyRoll > 70)
121  return GREATSUCCESS;
122 
123  if (assemblyRoll > 60)
124  return GOODSUCCESS;
125 
126  if (assemblyRoll > 50)
127  return MODERATESUCCESS;
128 
129  if (assemblyRoll > 40)
130  return SUCCESS;
131 
132  if (assemblyRoll > 30)
133  return MARGINALSUCCESS;
134 
135  if (assemblyRoll > 20)
136  return OK;
137 
138  return BARELYSUCCESSFUL;
139 }

262 bool CraftingManagerImplementation::loadBioSkillMods() {
263  Reference<Lua* > lua = new Lua();
264  lua->init();
265 
266  if (!lua->runFile("scripts/crafting/bio_skill_mods.lua")) {
267  return false;
268  }
269 
270  LuaObject bioModsTable = lua->getGlobalObject("bioSkillMods");
271 
272  if (!bioModsTable.isValidTable())
273  return false;
274 
275  for (int i = 1; i <= bioModsTable.getTableSize(); ++i) {
276  String mod = bioModsTable.getStringAt(i);
277  bioMods.put(mod);
278  }
279 
280  bioModsTable.pop();
281 
282  return true;
283 
284 }
