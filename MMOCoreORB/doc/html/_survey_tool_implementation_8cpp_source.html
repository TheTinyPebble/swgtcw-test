87 int SurveyToolImplementation::handleObjectMenuSelect(CreatureObject* player, byte selectedID) {
88  PlayerObject* playerObject = player->getPlayerObject();
89 
90  if(isASubChildOf(player)) {
91  if (!playerObject->hasAbility("survey")) {
92  player->sendSystemMessage("@error_message:insufficient_skill");
93  return 0;
94  }
95 
96  if (selectedID == 20) { // use object
97 
98  if(getRange(player) == 0) {
99  sendRangeSui(player);
100  return 0;
101  }
102 
103  Locker locker(_this.get());
104 
105  ManagedReference<SurveySession*> session = player->getActiveSession(SessionFacadeType::SURVEY).castTo<SurveySession*>();
106  if(session == NULL) {
107  session = new SurveySession(player);
108  session->initializeSession(_this.get());
109  }
110 
111  session->setOpenSurveyTool(_this.get());
112 
113  ManagedReference<ResourceManager*> resourceManager = cast<ResourceManager*>(server->getZoneServer()->getResourceManager());
114  if(resourceManager == NULL) {
115  error("null resource manager");
116  return 0;
117  }
118  resourceManager->sendResourceListForSurvey(player, getToolType(), getSurveyType());
119 
120  return 0;
121  }
122 
123  if (selectedID == 133) { // Set Tool Range
124 
125  sendRangeSui(player);
126 
127  return 0;
128  }
129  }
130 
131  return TangibleObjectImplementation::handleObjectMenuSelect(player, selectedID);
132 }
