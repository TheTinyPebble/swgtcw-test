93  void loadStateData() {
94  IffStream* iffStream = TemplateManager::instance()->openIffFile("datatables/include/state.iff");
95 
96  if (iffStream == NULL) {
97  error("Could not load states.");
98  return;
99  }
100 
101  DataTableIff dtiff;
102  dtiff.readObject(iffStream);
103 
104  delete iffStream;
105 
106  states.removeAll();
107 
108  for (int i = 0; i < dtiff.getTotalRows(); i++) {
109  DataTableRow* row = dtiff.getRow(i);
110  String name;
111  int value;
112 
113  row->getValue(0, name);
114  row->getValue(1, value);
115 
116  if (value >= 0)
117  states.put(name.toLowerCase(), 1 << value);
118  else
119  states.put(name.toLowerCase(), 0x00);
120  }
121  }
