97  // TODO: add shuttle jumping and planet jumping when path finding in cities is good enough.
98  void run() {
99  ManagedReference<BountyMissionObjective*> objectiveRef = objective.get();
100 
101  if (objectiveRef == NULL) {
102  return;
103  }
104 
105  ManagedReference<CreatureObject*> playerRef = player.get();
106 
107  if (playerRef == NULL || playerRef->getZoneServer() == NULL) {
108  return;
109  }
110 
111  Locker locker(playerRef);
112 
113  if (objectiveRef->getPlayerOwner() == NULL)
114  return;
115 
116  if (move) {
117  updatePosition(playerRef);
118  }
119 
120  ManagedReference<MissionObject*> strongMissionRef = objectiveRef->getMissionObject();
121 
122  String zoneName = strongMissionRef->getEndPlanet();
123 
124  Zone* zone = playerRef->getZone();
125 
126  if (zone != NULL && zone->getZoneName() == zoneName) {
127  Vector3 playerPosition = playerRef->getWorldPosition();
128  playerPosition.setZ(0);
129 
130  if (playerPosition.distanceTo(currentPosition) < 500.0f) {
131  updateToSpawnableTargetPosition();
132  if (playerPosition.distanceTo(currentPosition) < 500.0f) {
133  move = false;
134  objectiveRef->spawnTarget(zoneName);
135  }
136  }
137  }
138 
139  reschedule(10 * 1000);
140  }

150 private:
151  void updatePosition(CreatureObject* player) {
152  Vector3 direction = nextPosition - currentPosition;
153  Vector3 movementUpdate = direction;
154  movementUpdate.normalize();
155 
156  if (direction.length() > 10.0) {
157  currentPosition = currentPosition + (10 * movementUpdate);
158  } else {
159  currentPosition = nextPosition;
160  nextPosition = player->getZoneServer()->getMissionManager()->getRandomBountyTargetPosition(player);
161  nextPosition.setZ(0);
162  }
163  }
164 
165  void updateToSpawnableTargetPosition() {
166  if (canSpawnTargetAt(currentPosition)) {
167  return;
168  }
169 
170  //Spawning at coordinates failed, try to find new coordinates.
171  int radius = 50;
172  while (radius <= 1600) {
173  //Max 20 retries per radius, total 120 retries.
174  int retries = 20;
175 
176  while (retries > 0) {
177  //Generate a random direction and move the target position in that direction within the radius.
178  Vector3 direction;
179  direction.setX(System::random(2 * radius) - radius);
180  direction.setY(System::random(2 * radius) - radius);
181  direction.normalize();
182  direction = direction * System::random(radius);
183 
184  if (canSpawnTargetAt(currentPosition + direction)) {
185  currentPosition = currentPosition + direction;
186  return;
187  }
188 
189  retries--;
190  }
191 
192  radius *= 2;
193  }
194 
195  //Failed to find new spawn for the target, spawn at current position.
196  }
197 
198  bool canSpawnTargetAt(const Vector3& position) {
199  ManagedReference<CreatureObject*> playerRef = player.get();
200 
201  if (playerRef == NULL || playerRef->getZone() == NULL) {
202  return false;
203  }
204 
205  Zone* zone = playerRef->getZone();
206 
207  if (zone->isWithinBoundaries(position)) {
208  if (zone->getPlanetManager()->isBuildingPermittedAt(position.getX(), position.getY(), NULL) &&
209  !zone->getPlanetManager()->isInWater(position.getX(), position.getY())) {
210  return true;
211  }
212  }
213 
214  return true;
215  }
