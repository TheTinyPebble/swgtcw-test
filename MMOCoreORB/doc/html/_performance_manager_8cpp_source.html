205 void PerformanceManager::loadPerformances() {
206 
207  IffStream* iffStream = TemplateManager::instance()->openIffFile(
208  "datatables/performance/performance.iff");
209 
210  if (iffStream == NULL) {
211  error("Could not open performances datatable.");
212  return;
213  }
214 
215  DataTableIff dtable;
216  dtable.readObject(iffStream);
217 
218  delete iffStream;
219 
220  performances = new Vector<Performance*> ();
221  for (int i = 0; i < dtable.getTotalRows(); ++i) {
222  DataTableRow* row = dtable.getRow(i);
223 
224  Performance* performance = new Performance();
225  performance->parseDataTableRow(row);
226  performances->add(performance);
227  }
228 
229  info("Loaded " + String::valueOf(performances->size()) + " performances.",
230  true);
231 }

233 Vector<Performance*> PerformanceManager::getPerformanceListFromMod(
234  const String& requiredSkillMod, int playerSkillModValue, int instrument) {
235  String instrumentName = "";
236  if (instrument != 0)
237  instrumentName = getInstrument(instrument);
238 
239  Vector<Performance*> performanceList;
240 
241  if (performances != NULL) {
242 
243  for (int i = 0; i < performances->size(); ++i) {
244  Performance* perform = performances->get(i);
245  if (perform->getRequiredSkillMod() == requiredSkillMod
246  && perform->getRequiredSkillModValue()
247  <= playerSkillModValue) {
248  if (instrumentName != "") {
249  //Should be a music call, look only for performances with that instrument
250  if (instrumentName == perform->getRequiredInstrument())
251  performanceList.add(perform);
252  } else {
253  //Should be a dance call
254  performanceList.add(perform);
255  }
256  }
257  }
258 
259  }
260 
261  return performanceList;
262 }
