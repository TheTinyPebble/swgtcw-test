76  void completeMission(unsigned int missionType, int reward) {
77  switch (missionType) {
78  case MissionObject::BOUNTY:
79  numberOfCompletedMissionsBounty.increment();
80  creditsGeneratedFromMissionsBounty.add(reward);
81  break;
82  case MissionObject::CRAFTING:
83  numberOfCompletedMissionsCrafting.increment();
84  creditsGeneratedFromMissionsCrafting.add(reward);
85  break;
86  case MissionObject::DANCER:
87  numberOfCompletedMissionsDancer.increment();
88  creditsGeneratedFromMissionsDancer.add(reward);
89  break;
90  case MissionObject::DELIVER:
91  numberOfCompletedMissionsDeliver.increment();
92  creditsGeneratedFromMissionsDeliver.add(reward);
93  break;
94  case MissionObject::DESTROY:
95  numberOfCompletedMissionsDestroy.increment();
96  creditsGeneratedFromMissionsDestroy.add(reward);
97  break;
98  case MissionObject::HUNTING:
99  numberOfCompletedMissionsHunting.increment();
100  creditsGeneratedFromMissionsHunting.add(reward);
101  break;
102  case MissionObject::MUSICIAN:
103  numberOfCompletedMissionsMusician.increment();
104  creditsGeneratedFromMissionsMusician.add(reward);
105  break;
106  case MissionObject::RECON:
107  numberOfCompletedMissionsRecon.increment();
108  creditsGeneratedFromMissionsRecon.add(reward);
109  break;
110  case MissionObject::SURVEY:
111  numberOfCompletedMissionsSurvey.increment();
112  creditsGeneratedFromMissionsSurvey.add(reward);
113  break;
114  }
115  }
