79  void run() {
80  Locker playerLocker(player);
81 
82  try {
83  Reference<MeditateTask*> meditateTask = player->getPendingTask("meditate").castTo<MeditateTask*>();
84  int meditateMod = player->getSkillMod("meditate"); // get Meditate Skill Mod
85 
86  if (!player->isMeditating())
87  return;
88 
89  if (player->isBleeding() || player->isPoisoned() || player->isDiseased()) {
90  if (player->isBleeding() && meditateMod >= 15)
91  player->healDot(CreatureState::BLEEDING, (15 + (meditateMod / 3)));
92  else if (player->isPoisoned() && meditateMod >= 30)
93  player->healDot(CreatureState::POISONED, (15 + (meditateMod / 3)));
94  else if (player->isDiseased() && meditateMod >= 45)
95  player->healDot(CreatureState::DISEASED, (15 + (meditateMod / 3)));
96 
97  } else if (meditateMod >= 75) { // Meditate SkillMod +75 for wound Healing..
98 
99  StringIdChatParameter healParams;
100 
101  //Here we are checking to see which pools have wounds, and we add them to a vector...
102  Vector<uint8> woundedPools;
103  for (uint8 i = 0; i < 9; ++i) {
104  if (player->getWounds(i) > 0)
105  woundedPools.add(i);
106  }
107 
108  //Return without rescheduling because everything that can be healed has been?
109  if (woundedPools.size() <= 0)
110  return;
111 
112  int heal = 20 + System::random(10);
113 
114  if (meditateMod >= 100)
115  heal = 30 + System::random(20);
116 
117  // Select a random Attribute that has wounds...
118  uint8 pool = woundedPools.get(System::random(woundedPools.size() - 1));
119 
120  int wounds = player->getWounds(pool);
121 
122  //Cap the heal at the amount of wounds the creature has.
123  heal = MIN(wounds, heal);
124 
125  player->addWounds(pool, -heal, true);
126 
127  // Sending System healing Message (wounds)
128  healParams.setStringId("teraskasi", "prose_curewound"); // [meditation] Your %TO wounds heal by %DI points.
129  healParams.setTO(CreatureAttribute::getName(pool));
130  healParams.setDI(heal);
131  player->sendSystemMessage(healParams);
132  }
133 
134  if (meditateTask != NULL)
135  meditateTask->reschedule(5000);
136  else
137  meditateTask->schedule(5000);
138 
139  } catch (Exception& e) {
140  player->error("unreported exception caught in MeditateTask::activate");
141  }
142  }
143 
144 };
