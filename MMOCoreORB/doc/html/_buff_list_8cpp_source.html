99  ManagedReference<Buff*> ref = buff;
100 
101  uint32 buffcrc = buff->getBuffCRC();
102 
103  //Remove the old buff if it exists. (Exists checked in removeBuff)
104  if ((buff->getBuffType() == BuffType::FOOD && !buff->isAttributeBuff())
105  || (buff->getBuffType() != BuffType::FOOD))
106  removeBuff(buffcrc);
107 
108  if (!buff->isPersistent())

131 void BuffList::removeBuff(Buff* buff) {
132  if (buff == NULL)
133  return;
134 
135  ManagedReference<Buff*> ref = buff;
136 
137  uint32 buffcrc = buff->getBuffCRC();
138 
139  if (buffList.contains(buffcrc)) {
140  //This only gets called if the event has been scheduled and is not executing.
141  if (buff->isActive())
142  buff->clearBuffEvent();
143 
144  if (buff->isSpiceBuff())
145  spiceActive = false;
146 
147  buff->clearBuffEvent();
148 
149  buffList.drop(buffcrc);
150 
151  //Already null checked the buff.
152  buff->deactivate();
153 
154  if (buff->isPersistent())
155  ObjectManager::instance()->destroyObjectFromDatabase(buff->_getObjectID());
156  }
