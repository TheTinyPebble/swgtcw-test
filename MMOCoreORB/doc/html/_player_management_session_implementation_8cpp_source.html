173 void PlayerManagementSessionImplementation::parseBanDuration(const String& duration) {
174 
175  StringTokenizer tokenizer(duration);
176  int durationInSeconds = 0;
177 
178  try {
179 
180  while(tokenizer.hasMoreTokens()) {
181  String token;
182  tokenizer.getStringToken(token);
183 
184  if(token.length() < 2)
185  throw Exception();
186 
187  char timeframe = token.charAt(token.length() - 1);
188  token = token.subString(0, token.length() - 1);
189  int units = Integer::valueOf(token);
190  int seconds = 0;
191 
192  switch(timeframe) {
193  case 'y':
194  seconds = 60 * 60 * 24 * 365;
195  break;
196  case 'm':
197  seconds = 60 * 60 * 24 * 30;
198  break;
199  case 'd':
200  seconds = 60 * 60 * 24;
201  break;
202  case 'h':
203  seconds = 60 * 60;
204  break;
205  case 'M':
206  seconds = 60;
207  break;
208  default:
209  throw Exception();
210  break;
211  }
212 
213  durationInSeconds += (units * seconds);
214  }
215 
216  banExpiration = durationInSeconds + time(0);
217 
218  if(banReason.isEmpty())
219  sendBanReason();
220  else
221  showBanSummary();
222 
223  } catch(Exception& e) {
