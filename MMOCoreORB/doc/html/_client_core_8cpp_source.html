151 void ClientCore::handleCommands() {
152  while (true) {
153  try {
154  String command;
155 
156  Thread::sleep(500);
157 
158  continue;
159 
160  System::out << "> ";
161 
162  char line[256];
163  fgets(line, sizeof(line), stdin);
164 
165  command = line;
166  command = command.replaceFirst("\n", "");
167 
168  StringTokenizer tokenizer(command);
169  String firstToken;
170  tokenizer.getStringToken(firstToken);
171 
172  if (firstToken == "exit") {
173  for (int i = 0; i < zones.size(); ++i)
174  zones.get(i)->disconnect();
175 
176  return;
177  } else if (firstToken == "follow") {
178  String name;
179  tokenizer.finalToken(name);
180 
181  for (int i = 0; i < zones.size(); ++i)
182  zones.get(i)->follow(name);
183 
184  } else if (firstToken == "stopFollow") {
185  for (int i = 0; i < zones.size(); ++i)
186  zones.get(i)->stopFollow();
187  } else if (firstToken == "lurk") {
188  for (int i = 0; i < zones.size(); ++i)
189  zones.get(i)->lurk();
190  } else if (firstToken == "info") {
191  for (int i = 0; i < zones.size(); ++i) {
192  uint32 size = zones.get(i)->getObjectManager()->getObjectMapSize();
193  StringBuffer msg;
194  msg << "[ObjectManager" << i << "] size: " << size;
195 
196  info(msg.toString(), true);
197  }
198 
199  } else {
200  String args;
201  if (tokenizer.hasMoreTokens())
202  tokenizer.finalToken(args);
203 
204  for (int i = 0; i < zones.size(); ++i)
205  if (!zones.get(i)->doCommand(firstToken, args))
206  Logger::console.error("unknown command");
207  }
208  } catch (SocketException& e) {
209  System::out << "[ClientCore] " << e.getMessage();
210  } catch (ArrayIndexOutOfBoundsException& e) {
211  System::out << "[ClientCore] " << e.getMessage() << "\n";
212  e.printStackTrace();
213 
214  } catch (Exception& e) {
215  StringBuffer msg;
216  msg << "[ClientCore] Exception caught";
217  error(msg.toString());
218  error(e.getMessage());
219  e.printStackTrace();
220 
221  }
222  }
223 }
