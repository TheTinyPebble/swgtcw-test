78  void addItemToList(AuctionItem* ai) {
79  locationList.put(ai->getVendorUID());

113  void dumpItemInfoList() {
114  int ilSize = itemList.size();
115 
116  insertInt(ilSize);
117 
118  for (int i = 0; i < itemList.size(); i++) {
119  AuctionItem* il = itemList.get(i);
120 
121  int accessFee = 0;
122  ManagedReference<SceneObject*> vendor = player->getZoneServer()->getObject(il->getVendorID());
123  if(vendor != NULL && vendor->getParent() != NULL) {
124  ManagedReference<SceneObject*> parent = vendor->getRootParent();
125  if(parent->isBuildingObject()) {
126  BuildingObject* building = cast<BuildingObject*>(parent.get());
127  if(building != NULL)
128  accessFee = building->getAccessFee();
129  }
130  }
131 
132  insertLong(il->getAuctionedItemObjectID()); //item id
133  insertByte(i); // List item String number
134 
135  insertInt(il->getPrice()); //item cost.
136 
137  Time expireTime;
138  uint32 expire = il->getExpireTime() - expireTime.getMiliTime() / 1000;
139 
140  insertInt(expire);
141 
142  if (il->isAuction())
143  insertByte(0);
144  else
145  insertByte(1);
146 
147  insertShort(locationList.find(il->getVendorUID()));
148 
149  insertLong(il->getOwnerID()); // seller ID
150 
151  insertShort(locationList.find(il->getOwnerName()));
152 
153  if(il->isAuction() && il->getStatus() == AuctionItem::FORSALE) {
154  insertLong(0);
155  insertShort(locationList.find(""));
156  } else {
157  insertLong(il->getBuyerID()); // buyer ID
158  insertShort(locationList.find(il->getBidderName()));
159  }
160 
161  insertInt(il->getProxy()); // my proxy not implemented yet
162  insertInt(il->getPrice()); // my bid default to price
163 
164  insertInt(il->getItemType());
165 
166  //insertInt(il->getAuctionOptions()); // autionOptions 0x400 = Premium, 0x800 = withdraw
167  int additionalValues = 0;
168 
169  if (il->getOwnerID() == player->getObjectID() &&
170  (il->getStatus() == AuctionItem::FORSALE || il->getStatus() == AuctionItem::OFFERED)) {
171  additionalValues |= 0x800;
172  }
173 
174  insertInt(il->getAuctionOptions() | additionalValues);
175  //insertInt(10);
176 
177  insertInt(accessFee);
178  }
179  }
