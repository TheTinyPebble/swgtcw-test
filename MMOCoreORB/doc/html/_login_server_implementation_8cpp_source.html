164  try {

208 void LoginServerImplementation::populateGalaxyList() {
209  //Populate the galaxies list for the login server.
210  GalaxyList galaxies;
211  uint32 galaxyCount = galaxies.size();
212 
213  //In case we want to add the functionality to update the lists while the server is running...
214  if (enumClusterMessage != NULL) {
215  delete enumClusterMessage;
216  enumClusterMessage = NULL;
217  }
218 
219  if (clusterStatusMessage != NULL) {
220  delete enumClusterMessage;
221  enumClusterMessage = NULL;
222  }
223 
224  enumClusterMessage = new LoginEnumCluster(galaxyCount);
225  clusterStatusMessage = new LoginClusterStatus(galaxyCount);
226 
227  while (galaxies.next()) {
228  uint32 galaxyID = galaxies.getGalaxyID();
229 
230  String name;
231  galaxies.getGalaxyName(name);
232 
233  enumClusterMessage->addGalaxy(galaxyID, name);
234 
235  String address;
236  galaxies.getGalaxyAddress(address);
237 
238  clusterStatusMessage->addGalaxy(galaxyID, address, galaxies.getGalaxyPort(), galaxies.getGalaxyPingPort());
239  }
240 
241  enumClusterMessage->finish();
242 }
