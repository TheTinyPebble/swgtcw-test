71  void deactivateWoundTreatment(CreatureObject* creature) {
72  float modSkill = (float)creature->getSkillMod("healing_wound_speed");
73 
74  int delay = (int)round((modSkill * -(2.0f / 25.0f)) + 20.0f);
75 
76  if (creature->hasBuff(BuffCRC::FOOD_HEAL_RECOVERY)) {
77  DelayedBuff* buff = cast<DelayedBuff*>( creature->getBuff(BuffCRC::FOOD_HEAL_RECOVERY));
78 
79  if (buff != NULL) {
80  float percent = buff->getSkillModifierValue("heal_recovery");
81 
82  delay = round(delay * (100.0f - percent) / 100.0f);
83  }
84  }
85 
86  //Force the delay to be at least 3 seconds.
87  delay = (delay < 3) ? 3 : delay;
88 
89  StringIdChatParameter message("healing_response", "healing_response_59"); //You are now ready to heal more wounds or apply more enhancements.
90  Reference<InjuryTreatmentTask*> task = new InjuryTreatmentTask(creature, message, "woundTreatment");
91  creature->addPendingTask("woundTreatment", task, delay * 1000);
92  }

94  EnhancePack* findEnhancePack(CreatureObject* enhancer, uint8 attribute) {
95  SceneObject* inventory = enhancer->getSlottedObject("inventory");
96 
97  int medicineUse = enhancer->getSkillMod("healing_ability");
98 
99  if (inventory != NULL) {
100  for (int i = 0; i < inventory->getContainerObjectsSize(); ++i) {
101  SceneObject* object = inventory->getContainerObject(i);
102 
103  if (!object->isTangibleObject())
104  continue;
105 
106  TangibleObject* item = cast<TangibleObject*>( object);
107 
108  if (item->isPharmaceuticalObject()) {
109  PharmaceuticalObject* pharma = cast<PharmaceuticalObject*>( item);
110 
111  if (pharma->isEnhancePack()) {
112  EnhancePack* enhancePack = cast<EnhancePack*>( pharma);
113 
114  if (enhancePack->getMedicineUseRequired() <= medicineUse && enhancePack->getAttribute() == attribute)
115  return enhancePack;
116  }
117  }
118  }
119  }

121  return NULL;

124  bool canPerformSkill(CreatureObject* enhancer, CreatureObject* patient, EnhancePack* enhancePack) {
125  if (patient->isDead())

127 
128  if (!enhancer->canTreatWounds()) {
129  enhancer->sendSystemMessage("@healing_response:enhancement_must_wait"); //You must wait before you can heal wounds or apply enhancements again.

132 
133  if (enhancePack == NULL) {
134  enhancer->sendSystemMessage("@healing_response:healing_response_60"); //No valid medicine found.
135  return false;
136  }
137 
138  if (enhancer->getSkillMod("private_medical_rating") <= 0) {
139  enhancer->sendSystemMessage("@healing_response:must_be_near_droid"); //You must be in a hospital, at a campsite, or near a surgical droid to do that.
140  return false;
141  }
142 
143  if (enhancer->isProne() || enhancer->isMeditating()) {
144  enhancer->sendSystemMessage("@error_message:wrong_state"); //You cannot complete that action while in your current state.
145  return false;
146  }
147 
148  if (enhancer->isRidingCreature() || enhancer->isMounted()) {
149  enhancer->sendSystemMessage("@error_message:survey_on_mount"); //You cannot perform that action while mounted on a creature or driving a vehicle.
150  return false;
151  }
152 
153  if (enhancer->isInCombat()) {
154  enhancer->sendSystemMessage("You cannot do that while in Combat.");
155  return false;
156  }
157 
158  if (patient->isInCombat()) {
159  enhancer->sendSystemMessage("You cannot do that while your target is in Combat.");
160  return false;
161  }
162 
163  if (!patient->isHealableBy(enhancer)) {
164  enhancer->sendSystemMessage("@healing:pvp_no_help"); //It would be unwise to help such a patient.
165  return false;
166  }
167 
168  if (enhancer->getHAM(CreatureAttribute::MIND) < mindCost) {
169  enhancer->sendSystemMessage("@healing_response:not_enough_mind"); //You do not have enough mind to do that.
170  return false;
171  }
172 
173  if (enhancer != patient && !CollisionManager::checkLineOfSight(enhancer, patient)) {
174  enhancer->sendSystemMessage("@container_error_message:container18");
175  return false;
176  }
177 
178  return true;
179  }

181  void parseModifier(const String& modifier, uint8& attribute, uint64& objectId) {
182  if (!modifier.isEmpty()) {

184  tokenizer.setDelimeter("|");

186  String attributeName;
187 
188  tokenizer.getStringToken(attributeName);
189  attribute = BuffAttribute::getAttribute(attributeName);
190 
191  if (tokenizer.hasMoreTokens())
192  objectId = tokenizer.getLongToken();
193  } else {
194  attribute = BuffAttribute::UNKNOWN;
195  objectId = 0;

269  int doQueueCommand(CreatureObject* creature, const uint64& target, const UnicodeString& arguments) {
270 
271  if (!checkStateMask(creature))
272  return INVALIDSTATE;
273 
274  if (!checkInvalidLocomotions(creature))
275  return INVALIDLOCOMOTION;
276 
277  CreatureObject* enhancer = creature;
278 
279  ManagedReference<SceneObject*> object = server->getZoneServer()->getObject(target);
280 
281  if (object != NULL) {
282  if (!object->isCreatureObject()) {
283  TangibleObject* tangibleObject = dynamic_cast<TangibleObject*>(object.get());
284 
285  if (tangibleObject != NULL && tangibleObject->isAttackableBy(creature)) {
286  object = creature;
287  } else
288  creature->sendSystemMessage("@healing_response:healing_response_77"); //Target must be a player or a creature pet in order to apply enhancements.
289  return GENERALERROR;
290  }
291  } else
292  object = creature;
293 
294  CreatureObject* targetCreature = cast<CreatureObject*>( object.get());
295 
296  uint8 attribute = BuffAttribute::UNKNOWN;
297  uint64 objectId = 0;
298 
299  if (!targetCreature->isInRange(creature, range))
300  return TOOFAR;
301 
302  parseModifier(arguments.toString(), attribute, objectId);
303 
304  if (attribute == BuffAttribute::UNKNOWN) {
305  enhancer->sendSystemMessage("@healing_response:healing_response_75"); //You must specify a valid attribute.
306  return GENERALERROR;
307  }
308 
309  ManagedReference<EnhancePack*> enhancePack = NULL;
310 
311  if (objectId != 0) {
312  SceneObject* inventory = creature->getSlottedObject("inventory");
313 
314  if (inventory != NULL) {
315  enhancePack = inventory->getContainerObject(objectId).castTo<EnhancePack*>();
316  }
317  } else {
318  enhancePack = findEnhancePack(creature, attribute);
319  }
320 
321  CreatureObject* patient = cast<CreatureObject*>( targetCreature);
322 
323  Locker clocker(patient, creature);
324 
325  if (patient->isDead() || patient->isRidingCreature() || patient->isMounted())
326  patient = enhancer;
327 
328  if (!canPerformSkill(enhancer, patient, enhancePack))
329  return GENERALERROR;
330 
331  uint32 currentBuff = 0;
332  uint32 buffcrc = BuffCRC::getMedicalBuff(attribute);
333 
334  if (patient->hasBuff(buffcrc)) {
335  Buff* existingbuff = patient->getBuff(buffcrc);
336 
337  if (existingbuff != NULL){
338 
339  if (BuffAttribute::isProtection(attribute))
340  currentBuff = existingbuff->getSkillModifierValue(BuffAttribute::getProtectionString(attribute));
341  else
342  currentBuff = existingbuff->getAttributeModifierValue(attribute);
343  }
344  }
345 
346  //Applies battle fatigue
347  uint32 buffPower = 0;
348  if (BuffAttribute::isProtection(attribute)) { // If it's a protection enhancement, wound treatment has no effect
349  buffPower = enhancePack->getEffectiveness();
350  buffPower = buffPower * (1 - patient->calculateBFRatio()) * (1 - enhancer->calculateBFRatio());
351  } else
352  buffPower = enhancePack->calculatePower(enhancer, patient);
353 
354  if (buffPower < currentBuff) {
355  if (patient == enhancer)
356  enhancer->sendSystemMessage("Your current enhancements are of greater power and cannot be re-applied.");
357  else
358  enhancer->sendSystemMessage("Your target's current enhancements are of greater power and cannot be re-applied.");
359 
360  return 0;
361  }
362 
363  PlayerManager* playerManager = server->getZoneServer()->getPlayerManager();
364 
365  uint32 amountEnhanced = playerManager->healEnhance(enhancer, patient, attribute, buffPower, enhancePack->getDuration());
366 
367  if (creature->isPlayerCreature() && targetCreature->isPlayerCreature()) {
368  playerManager->sendBattleFatigueMessage(creature, targetCreature);
369  }
370 
371  sendEnhanceMessage(enhancer, patient, attribute, amountEnhanced);
372 
373  enhancer->inflictDamage(enhancer, CreatureAttribute::MIND, mindCost, false);
374 
375  deactivateWoundTreatment(enhancer);
376 
377  if (enhancePack != NULL)
378  enhancePack->decreaseUseCount();
379 
380  if (patient != enhancer)
381  awardXp(enhancer, "medical", amountEnhanced); //No experience for healing yourself.
382 
383  doAnimations(enhancer, patient);
384 
385  creature->notifyObservers(ObserverEventType::MEDPACKUSED);
386 
387  return SUCCESS;
388  }
