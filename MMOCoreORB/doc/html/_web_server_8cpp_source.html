124 void WebServer::whitelistInit() {
125 
126  authorizedUsers.removeAll();
127  authorizedIpAddresses.removeAll();
128 
129  info("Parsing access whitelist 'conf/webusers.lst'", true);
130 
131  if(zoneServer == NULL) {
132  error("Zone server lookup failed, unable to verify users for access");
133  return;
134  }
135 
136  File* webusersFile = new File("conf/webusers.lst");
137 
138  try {
139  FileReader webusersReader(webusersFile);
140 
141  String line, username, ipaddress;
142 
143  while (webusersReader.readLine(line)) {
144  String entry = line.trim().toLowerCase();
145 
146  if(entry.indexOf("--") != -1)
147  entry = entry.subString(0, entry.indexOf("--")).trim();
148 
149  if(entry.isEmpty())
150  continue;
151 
152  StringTokenizer tokenizer(entry);
153  tokenizer.setDelimeter(";");
154 
155  if(tokenizer.hasMoreTokens()) {
156  tokenizer.getStringToken(username);
157  } else {
158  error("Whitelist entry error for " + line);
159  continue;
160  }
161 
162  if(tokenizer.hasMoreTokens()) {
163  tokenizer.getStringToken(ipaddress);
164  } else {
165  error("Whitelist ip error for " + username);
166  continue;
167  }
168 
169  if(isValidIp(ipaddress)) {
170 
171  authorizedIpAddresses.add(ipaddress);
172 
173  WebCredentials* credentials;
174 
175  if(authorizedUsers.contains(username)) {
176  credentials = authorizedUsers.get(username);
177  credentials->addInetAddress(ipaddress);
178  } else {
179  credentials = new WebCredentials(username, ipaddress);
180  authorizedUsers.put(username, credentials);
181  }
182  } else {
183  error("IP address is invalid for: " + username + " IP: " + ipaddress);
184  }
185  }
186  } catch (FileNotFoundException&e ) {
187 
188  error("Whitelist not found");
189 
190  }
191 
192  delete webusersFile;
193 }

345  session = NULL;
346  }

353  activeSessions.drop(request_info->remote_ip);

356  }

358  if(session != NULL)
359  session->update(conn, request_info);
360 
361  }
362 
363  if (session == NULL) {
364 
365  session = new HttpSession();
366  session->update(conn, request_info);
367  session->setSessionIp(request_info->remote_ip);

370  if (sessionList == NULL) {
371  sessionList = new HttpSessionList();
372  activeSessions.put(request_info->remote_ip, sessionList);

376 
377  if(!requestSessionId.isEmpty()) {

379  }
380 
381  sessionList->put(session->getSessionId(), session);
382 
383  }
384 
386  int length = Integer::valueOf(session->getRequest()->getHeader("Content-Length")) + 1;
387 
389  if(length > 0 && length < 1024) {
390

392 
393  mg_read(conn, postDataBuffer, 1024);
394 
395  postDataBuffer[length - 1] = '\0';
396 
397  session->getRequest()->updatePostData(postDataBuffer);
398 
399  delete [] postDataBuffer;
400 
401  } else if(length >= 1000) {

535 uint32 WebServer::ipStringToLong(String address) {
536 
537  long returnAddress = 0;
538  int tokencount = 0;
539 
540  StringTokenizer ipAddress(address);
541  ipAddress.setDelimeter(".");
542 
543 
544  while(ipAddress.hasMoreTokens()) {
545  try {
546 
547  int octet = ipAddress.getIntToken();
548  tokencount++;
549 
550  if(tokencount == 5)
551  return 0;
552 
553  if(octet >= 0 && octet <= 255) {
554 
555  returnAddress += (octet << (32 - (tokencount * 8)));
556 
557  } else {
558  return 0;
559  }
560 
561  } catch (const Exception& e) {
562  return 0;
563  }
564  }
565 
566  if(tokencount == 4)
567  return returnAddress;
568  else
569  return 0;
570 }
