61 class CallArakydTask : public Task, public Logger {
62  ManagedWeakReference<CreatureObject*> player;
63  ManagedWeakReference<BountyMissionObjective*> objective;
64  int time;
65  Vector3 droidPosition;
66 
67 public:
68  CallArakydTask(CreatureObject* player, BountyMissionObjective* objective) :
69  Logger("FindTargetTask") {
70  this->player = player;
71  this->objective = objective;
72  time = 20;
73  }
74 
75  ~CallArakydTask() {
76  cancel();
77  }

79  void run() {
80  ManagedReference<CreatureObject*> playerRef = player.get();
81  ManagedReference<BountyMissionObjective*> objectiveRef = objective.get();
82 
83  if (playerRef == NULL || objectiveRef == NULL) {
84  return;
85  }
86 
87  switch (time) {
88  case 20:
89  playerRef->sendSystemMessage("@mission/mission_generic:probe_droid_launch_prep");
90  time -= 15;
91  reschedule(15 * 1000);
92  break;
93  case 3: {
94  droidPosition = getLandingCoordinates(playerRef);
95  PlayClientEffectLoc* effect = new PlayClientEffectLoc("clienteffect/probot_delivery.cef", playerRef->getZone()->getZoneName(), droidPosition.getX(), droidPosition.getZ(), droidPosition.getY(), 0, 0);
96  playerRef->sendMessage(effect);
97  }
98  case 5:
99  case 4:
100  case 2:
101  case 1:
102  playerRef->sendSystemMessage("@mission/mission_generic:probe_droid_arrival_" + String::valueOf(time));
103  time -= 1;
104  reschedule(1 * 1000);
105  break;
106  case 0: {
107  playerRef->sendSystemMessage("@mission/mission_generic:probe_droid_arrival");
108  ManagedReference<AiAgent*> droid = cast<AiAgent*>(playerRef->getZone()->getCreatureManager()->spawnCreature(String("probot").hashCode(), 0, droidPosition.getX(), droidPosition.getZ(), droidPosition.getY(), 0));
109  objectiveRef->setArakydDroid(droid);
110  droid->setFollowObject(playerRef);
111  }
112  break;
113  default:
114  error("Unknowns state.");
115  break;
116  }
117  }

119  Vector3 getLandingCoordinates(CreatureObject* player) {
120  Vector3 position = player->getPosition();
121 
122  if (player->getZone() == NULL || player->getZone()->getPlanetManager() == NULL) {
123  return position;
124  }
125 
126  PlanetManager* planetManager = player->getZone()->getPlanetManager();
127 
128  int distance = 30;
129  int angle = 15;
130 
131  do {
132  for (int i = 0; i < 10; i++) {
133  position = player->getWorldCoordinate(distance + System::random(20), angle - System::random(2 * angle));
134 
135  if (noInterferingObjects(player, position)) {
136  return position;
137  }
138  }
139 
140  distance += 10;
141  angle += 5;
142  } while (distance <= 60);
143 
144  return player->getPosition();
145  }
146 
147  bool noInterferingObjects(CreatureObject* player, Vector3 position)
148  {
149  SortedVector<ManagedReference<QuadTreeEntry* > >* closeObjects = player->getCloseObjects();
150 
151  for (int j = 0; j < closeObjects->size(); j++) {
152  SceneObject* obj = cast<SceneObject*>(closeObjects->get(j).get());
153 
154  SharedObjectTemplate* objectTemplate = obj->getObjectTemplate();
155 
156  if (objectTemplate != NULL) {
157  float radius = objectTemplate->getNoBuildRadius();
158 
159  if (radius > 0) {
160  Vector3 objWorldPos = obj->getWorldPosition();
161 
162  if (objWorldPos.squaredDistanceTo(position) < radius * radius) {
163  return false;
164  }
165  }
166  }
167  }
168 
169  return true;
170  }
