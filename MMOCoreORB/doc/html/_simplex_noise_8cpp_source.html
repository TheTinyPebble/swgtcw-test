95 /*
96  * Helper functions to compute gradients-dot-residualvectors (1D to 4D)
97  * Note that these generate gradients of more than unit length. To make
98  * a close match with the value range of classic Perlin noise, the final
99  * noise values need to be rescaled to fit nicely within [-1,1].
100  * (The simplex noise functions as such also have different scaling.)
101  * Note also that these noise functions are the most practical and useful
102  * signed version of Perlin noise. To return values according to the
103  * RenderMan specification from the SL noise() and pnoise() functions,
104  * the noise values need to be scaled and offset to [0,1], like this:
105  * float SLnoise = (SimplexNoise::noise(x,y,z) + 1.0) * 0.5;
106  */

108 float SimplexNoise::grad( int hash, float x ) {
109  int h = hash & 15;
110  float grad = 1.0f + (h & 7); // Gradient value 1.0, 2.0, ..., 8.0
111  if (h&8) grad = -grad; // Set a random sign for the gradient
112  return ( grad * x ); // Multiply the gradient with the distance
113 }

115 float SimplexNoise::grad( int hash, float x, float y ) {
116  int h = hash & 7; // Convert low 3 bits of hash code
117  float u = h<4 ? x : y; // into 8 simple gradient directions,
118  float v = h<4 ? y : x; // and compute the dot product with (x,y).
119  return ((h&1)? -u : u) + ((h&2)? -2.0f*v : 2.0f*v);
120 }

122 float SimplexNoise::grad( int hash, float x, float y , float z ) {
123  int h = hash & 15; // Convert low 4 bits of hash code into 12 simple
124  float u = h<8 ? x : y; // gradient directions, and compute dot product.
125  float v = h<4 ? y : h==12||h==14 ? x : z; // Fix repeats at h = 12 to 15
126  return ((h&1)? -u : u) + ((h&2)? -v : v);
127 }

129 float SimplexNoise::grad( int hash, float x, float y, float z, float t ) {
130  int h = hash & 31; // Convert low 5 bits of hash code into 32 simple
131  float u = h<24 ? x : y; // gradient directions, and compute dot product.
132  float v = h<16 ? y : z;
133  float w = h<8 ? z : t;
134  return ((h&1)? -u : u) + ((h&2)? -v : v) + ((h&4)? -w : w);

150 // 1D simplex noise
151 float SimplexNoise::noise(float x) {
152 
153  int i0 = FASTFLOOR(x);
154  int i1 = i0 + 1;
155  float x0 = x - i0;
156  float x1 = x0 - 1.0f;
157 
158  float n0, n1;
159 
160  float t0 = 1.0f - x0*x0;
161 // if(t0 < 0.0f) t0 = 0.0f;
162  t0 *= t0;
163  n0 = t0 * t0 * grad(perm[i0 & 0xff], x0);
164 
165  float t1 = 1.0f - x1*x1;
166 // if(t1 < 0.0f) t1 = 0.0f;
167  t1 *= t1;
168  n1 = t1 * t1 * grad(perm[i1 & 0xff], x1);
169  // The maximum value of this noise is 8*(3/4)^4 = 2.53125
170  // A factor of 0.395 would scale to fit exactly within [-1,1], but
171  // we want to match PRMan's 1D noise, so we scale it down some more.
172  return 0.25f * (n0 + n1);
173 
174 }

176 // 2D simplex noise
177 float SimplexNoise::noise(float x, float y) {
178 
179 #define F2 0.366025403 // F2 = 0.5*(sqrt(3.0)-1.0)
180 #define G2 0.211324865 // G2 = (3.0-Math.sqrt(3.0))/6.0
181 
182  float n0, n1, n2; // Noise contributions from the three corners
183 
184  // Skew the input space to determine which simplex cell we're in
185  float s = (x+y)*F2; // Hairy factor for 2D
186  float xs = x + s;
187  float ys = y + s;
188  int i = FASTFLOOR(xs);
189  int j = FASTFLOOR(ys);
190 
191  float t = (float)(i+j)*G2;
192  float X0 = i-t; // Unskew the cell origin back to (x,y) space
193  float Y0 = j-t;
194  float x0 = x-X0; // The x,y distances from the cell origin
195  float y0 = y-Y0;
196 
197  // For the 2D case, the simplex shape is an equilateral triangle.
198  // Determine which simplex we are in.
199  int i1, j1; // Offsets for second (middle) corner of simplex in (i,j) coords
200  if(x0>y0) {i1=1; j1=0;} // lower triangle, XY order: (0,0)->(1,0)->(1,1)
201  else {i1=0; j1=1;} // upper triangle, YX order: (0,0)->(0,1)->(1,1)
202 
203  // A step of (1,0) in (i,j) means a step of (1-c,-c) in (x,y), and
204  // a step of (0,1) in (i,j) means a step of (-c,1-c) in (x,y), where
205  // c = (3-sqrt(3))/6
206 
207  float x1 = x0 - i1 + G2; // Offsets for middle corner in (x,y) unskewed coords
208  float y1 = y0 - j1 + G2;
209  float x2 = x0 - 1.0f + 2.0f * G2; // Offsets for last corner in (x,y) unskewed coords
210  float y2 = y0 - 1.0f + 2.0f * G2;
211 
212  // Wrap the integer indices at 256, to avoid indexing perm[] out of bounds
213  int ii = i % 256;
214  int jj = j % 256;
215 
216  // Calculate the contribution from the three corners
217  float t0 = 0.5f - x0*x0-y0*y0;
218  if(t0 < 0.0f) n0 = 0.0f;
219  else {
220  t0 *= t0;
221  n0 = t0 * t0 * grad(perm[ii+perm[jj]], x0, y0);
222  }
223 
224  float t1 = 0.5f - x1*x1-y1*y1;
225  if(t1 < 0.0f) n1 = 0.0f;
226  else {
227  t1 *= t1;
228  n1 = t1 * t1 * grad(perm[ii+i1+perm[jj+j1]], x1, y1);
229  }
230 
231  float t2 = 0.5f - x2*x2-y2*y2;
232  if(t2 < 0.0f) n2 = 0.0f;
233  else {
234  t2 *= t2;
235  n2 = t2 * t2 * grad(perm[ii+1+perm[jj+1]], x2, y2);
236  }
237 
238  // Add contributions from each corner to get the final noise value.
239  // The result is scaled to return values in the interval [-1,1].
240  return 40.0f * (n0 + n1 + n2); // TODO: The scale factor is preliminary!
241  }

243 // 3D simplex noise
244 float SimplexNoise::noise(float x, float y, float z) {
245 
246 // Simple skewing factors for the 3D case
247 #define F3 0.333333333
248 #define G3 0.166666667
249 
250  float n0, n1, n2, n3; // Noise contributions from the four corners
251 
252  // Skew the input space to determine which simplex cell we're in
253  float s = (x+y+z)*F3; // Very nice and simple skew factor for 3D
254  float xs = x+s;
255  float ys = y+s;
256  float zs = z+s;
257  int i = FASTFLOOR(xs);
258  int j = FASTFLOOR(ys);
259  int k = FASTFLOOR(zs);
260 
261  float t = (float)(i+j+k)*G3;
262  float X0 = i-t; // Unskew the cell origin back to (x,y,z) space
263  float Y0 = j-t;
264  float Z0 = k-t;
265  float x0 = x-X0; // The x,y,z distances from the cell origin
266  float y0 = y-Y0;
267  float z0 = z-Z0;
268 
269  // For the 3D case, the simplex shape is a slightly irregular tetrahedron.
270  // Determine which simplex we are in.
271  int i1, j1, k1; // Offsets for second corner of simplex in (i,j,k) coords
272  int i2, j2, k2; // Offsets for third corner of simplex in (i,j,k) coords
273 
274 /* This code would benefit from a backport from the GLSL version! */
275  if(x0>=y0) {
276  if(y0>=z0)
277  { i1=1; j1=0; k1=0; i2=1; j2=1; k2=0; } // X Y Z order
278  else if(x0>=z0) { i1=1; j1=0; k1=0; i2=1; j2=0; k2=1; } // X Z Y order
279  else { i1=0; j1=0; k1=1; i2=1; j2=0; k2=1; } // Z X Y order
280  }
281  else { // x0<y0
282  if(y0<z0) { i1=0; j1=0; k1=1; i2=0; j2=1; k2=1; } // Z Y X order
283  else if(x0<z0) { i1=0; j1=1; k1=0; i2=0; j2=1; k2=1; } // Y Z X order
284  else { i1=0; j1=1; k1=0; i2=1; j2=1; k2=0; } // Y X Z order
285  }
286 
287  // A step of (1,0,0) in (i,j,k) means a step of (1-c,-c,-c) in (x,y,z),
288  // a step of (0,1,0) in (i,j,k) means a step of (-c,1-c,-c) in (x,y,z), and
289  // a step of (0,0,1) in (i,j,k) means a step of (-c,-c,1-c) in (x,y,z), where
290  // c = 1/6.
291 
292  float x1 = x0 - i1 + G3; // Offsets for second corner in (x,y,z) coords
293  float y1 = y0 - j1 + G3;
294  float z1 = z0 - k1 + G3;
295  float x2 = x0 - i2 + 2.0f*G3; // Offsets for third corner in (x,y,z) coords
296  float y2 = y0 - j2 + 2.0f*G3;
297  float z2 = z0 - k2 + 2.0f*G3;
298  float x3 = x0 - 1.0f + 3.0f*G3; // Offsets for last corner in (x,y,z) coords
299  float y3 = y0 - 1.0f + 3.0f*G3;
300  float z3 = z0 - 1.0f + 3.0f*G3;
301 
302  // Wrap the integer indices at 256, to avoid indexing perm[] out of bounds
303  int ii = i % 256;
304  int jj = j % 256;
305  int kk = k % 256;
306 
307  // Calculate the contribution from the four corners
308  float t0 = 0.6f - x0*x0 - y0*y0 - z0*z0;
309  if(t0 < 0.0f) n0 = 0.0f;
310  else {
311  t0 *= t0;
312  n0 = t0 * t0 * grad(perm[ii+perm[jj+perm[kk]]], x0, y0, z0);
313  }
314 
315  float t1 = 0.6f - x1*x1 - y1*y1 - z1*z1;
316  if(t1 < 0.0f) n1 = 0.0f;
317  else {
318  t1 *= t1;
319  n1 = t1 * t1 * grad(perm[ii+i1+perm[jj+j1+perm[kk+k1]]], x1, y1, z1);
320  }
321 
322  float t2 = 0.6f - x2*x2 - y2*y2 - z2*z2;
323  if(t2 < 0.0f) n2 = 0.0f;
324  else {
325  t2 *= t2;
326  n2 = t2 * t2 * grad(perm[ii+i2+perm[jj+j2+perm[kk+k2]]], x2, y2, z2);
327  }
328 
329  float t3 = 0.6f - x3*x3 - y3*y3 - z3*z3;
330  if(t3<0.0f) n3 = 0.0f;
331  else {
332  t3 *= t3;
333  n3 = t3 * t3 * grad(perm[ii+1+perm[jj+1+perm[kk+1]]], x3, y3, z3);
334  }
335 
336  // Add contributions from each corner to get the final noise value.
337  // The result is scaled to stay just inside [-1,1]
338  return 32.0f * (n0 + n1 + n2 + n3); // TODO: The scale factor is preliminary!
339  }

342 // 4D simplex noise
343 float SimplexNoise::noise(float x, float y, float z, float w) {
344 
345  // The skewing and unskewing factors are hairy again for the 4D case
346 #define F4 0.309016994 // F4 = (Math.sqrt(5.0)-1.0)/4.0
347 #define G4 0.138196601 // G4 = (5.0-Math.sqrt(5.0))/20.0
348 
349  float n0, n1, n2, n3, n4; // Noise contributions from the five corners
350 
351  // Skew the (x,y,z,w) space to determine which cell of 24 simplices we're in
352  float s = (x + y + z + w) * F4; // Factor for 4D skewing
353  float xs = x + s;
354  float ys = y + s;
355  float zs = z + s;
356  float ws = w + s;
357  int i = FASTFLOOR(xs);
358  int j = FASTFLOOR(ys);
359  int k = FASTFLOOR(zs);
360  int l = FASTFLOOR(ws);
361 
362  float t = (i + j + k + l) * G4; // Factor for 4D unskewing
363  float X0 = i - t; // Unskew the cell origin back to (x,y,z,w) space
364  float Y0 = j - t;
365  float Z0 = k - t;
366  float W0 = l - t;
367 
368  float x0 = x - X0; // The x,y,z,w distances from the cell origin
369  float y0 = y - Y0;
370  float z0 = z - Z0;
371  float w0 = w - W0;
372 
373  // For the 4D case, the simplex is a 4D shape I won't even try to describe.
374  // To find out which of the 24 possible simplices we're in, we need to
375  // determine the magnitude ordering of x0, y0, z0 and w0.
376  // The method below is a good way of finding the ordering of x,y,z,w and
377  // then find the correct traversal order for the simplex were in.
378  // First, six pair-wise comparisons are performed between each possible pair
379  // of the four coordinates, and the results are used to add up binary bits
380  // for an integer index.
381  int c1 = (x0 > y0) ? 32 : 0;
382  int c2 = (x0 > z0) ? 16 : 0;
383  int c3 = (y0 > z0) ? 8 : 0;
384  int c4 = (x0 > w0) ? 4 : 0;
385  int c5 = (y0 > w0) ? 2 : 0;
386  int c6 = (z0 > w0) ? 1 : 0;
387  int c = c1 + c2 + c3 + c4 + c5 + c6;
388 
389  int i1, j1, k1, l1; // The integer offsets for the second simplex corner
390  int i2, j2, k2, l2; // The integer offsets for the third simplex corner
391  int i3, j3, k3, l3; // The integer offsets for the fourth simplex corner
392 
393  // simplex[c] is a 4-vector with the numbers 0, 1, 2 and 3 in some order.
394  // Many values of c will never occur, since e.g. x>y>z>w makes x<z, y<w and x<w
395  // impossible. Only the 24 indices which have non-zero entries make any sense.
396  // We use a thresholding to set the coordinates in turn from the largest magnitude.
397  // The number 3 in the "simplex" array is at the position of the largest coordinate.
398  i1 = simplex[c][0]>=3 ? 1 : 0;
399  j1 = simplex[c][1]>=3 ? 1 : 0;
400  k1 = simplex[c][2]>=3 ? 1 : 0;
401  l1 = simplex[c][3]>=3 ? 1 : 0;
402  // The number 2 in the "simplex" array is at the second largest coordinate.
403  i2 = simplex[c][0]>=2 ? 1 : 0;
404  j2 = simplex[c][1]>=2 ? 1 : 0;
405  k2 = simplex[c][2]>=2 ? 1 : 0;
406  l2 = simplex[c][3]>=2 ? 1 : 0;
407  // The number 1 in the "simplex" array is at the second smallest coordinate.
408  i3 = simplex[c][0]>=1 ? 1 : 0;
409  j3 = simplex[c][1]>=1 ? 1 : 0;
410  k3 = simplex[c][2]>=1 ? 1 : 0;
411  l3 = simplex[c][3]>=1 ? 1 : 0;
412  // The fifth corner has all coordinate offsets = 1, so no need to look that up.
413 
414  float x1 = x0 - i1 + G4; // Offsets for second corner in (x,y,z,w) coords
415  float y1 = y0 - j1 + G4;
