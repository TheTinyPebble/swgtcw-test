496 void ZoneImplementation::updateCityRegions() {
497  info("updating " + String::valueOf(cityRegionUpdateVector.size()) + " cities", true);
498 
499  for (int i = 0; i < cityRegionUpdateVector.size(); ++i) {
500  CityRegion* city = cityRegionUpdateVector.get(i);
501 
502  Time* nextUpdateTime = city->getNextUpdateTime();
503  int seconds = -1 * round(nextUpdateTime->miliDifference() / 1000.f);
504 
505  if (seconds < 0) //If the update occurred in the past, force an immediate update.
506  seconds = 0;
507 
508  city->setLoaded();
509 
510  city->cleanupCitizens();
511 
512  city->rescheduleUpdateEvent(seconds);
513 
514  if (!city->isRegistered())
515  continue;
516 
517  if (city->getRegionsCount() == 0)
518  continue;
519 
520  Region* region = city->getRegion(0);
521 
522  unregisterObjectWithPlanetaryMap(region);
523  registerObjectWithPlanetaryMap(region);
524 
525  for(int i = 0; i < city->getStructuresCount(); i++){
526  ManagedReference<StructureObject*> structure = city->getCivicStructure(i);
527  unregisterObjectWithPlanetaryMap(structure);
528  registerObjectWithPlanetaryMap(structure);
529  }
530 
531  for(int i = 0; i < city->getCommercialStructuresCount(); i++){
532  ManagedReference<StructureObject*> structure = city->getCommercialStructure(i);
533  unregisterObjectWithPlanetaryMap(structure);
534  registerObjectWithPlanetaryMap(structure);
535  }
536  }
537 
538  cityRegionUpdateVector.removeAll();
539 }

541 bool ZoneImplementation::isWithinBoundaries(const Vector3& position) {
542  //Remove 1/16th of the size to match client limits. NOTE: it has not been verified to work like this in the client.
543  //Normal zone size is 8192, 1/16th of that is 512 resulting in 7680 as the boundary value.

546  float maxY = getMaxY() * 15 / 16;
547  float minY = getMinY() * 15 / 16;
548 
549  if (maxX >= position.getX() && minX <= position.getX() &&
550  maxY >= position.getY() && minY <= position.getY()) {
551  return true;
552  } else {
553  return false;
554  }
