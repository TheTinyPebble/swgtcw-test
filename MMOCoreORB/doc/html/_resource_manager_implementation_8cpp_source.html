323 uint32 ResourceManagerImplementation::getAvailablePowerFromPlayer(CreatureObject* player) {
324  SceneObject* inventory = player->getSlottedObject("inventory");
325  uint32 power = 0;
326 
327  for (int i = 0; i < inventory->getContainerObjectsSize(); i++) {
328  Reference<SceneObject*> obj = inventory->getContainerObject(i).castTo<SceneObject*>();
329 
330  if (obj == NULL || !obj->isResourceContainer())
331  continue;
332 
333  ResourceContainer* rcno = cast<ResourceContainer*>( obj.get());
334  ManagedReference<ResourceSpawn*> spawn = rcno->getSpawnObject();
335 
336  if (spawn == NULL || !spawn->isEnergy())
337  continue;
338 
339  int quantity = rcno->getQuantity();
340  int pe = spawn->getValueOf(CraftingManager::PE); // potential energy
341 
342  float modifier = MAX(1.0f, pe / 500.0f);
343 
344  power += (uint32) (modifier * quantity);
345  }
346 
347  return power;
348 }

350 void ResourceManagerImplementation::removePowerFromPlayer(CreatureObject* player, uint32 power) {
351  if (power == 0)
352  return;
353 
354  SceneObject* inventory = player->getSlottedObject("inventory");
355 
356  uint32 containerPower = 0;
357 
358  for (int i = 0; i < inventory->getContainerObjectsSize() && power > 0; ++i) {
359  ManagedReference<SceneObject*> obj = inventory->getContainerObject(i);
360 
361  if (obj == NULL || !obj->isResourceContainer())
362  continue;
363 
364  ResourceContainer* rcno = cast<ResourceContainer*>( obj.get());
365  ManagedReference<ResourceSpawn*> spawn = rcno->getSpawnObject();
366 
367  if (spawn == NULL || !spawn->isEnergy())
368  continue;
369 
370  int quantity = rcno->getQuantity();
371  int pe = spawn->getValueOf(CraftingManager::PE); // potential energy
372 
373  float modifier = MAX(1.0f, pe / 500.0f);
374 
375  containerPower = modifier * quantity;
376 
377  if (containerPower > power) {
378  uint32 consumedUnits = (uint64) power / modifier;
379  rcno->setQuantity(quantity - consumedUnits);
380 
381  ResourceContainerObjectDeltaMessage3* drcno3 = new ResourceContainerObjectDeltaMessage3(rcno);
382  drcno3->updateQuantity();
383  drcno3->close();
384  player->sendMessage(drcno3);
385  return;
386  } else {
387  rcno->destroyObjectFromWorld(true);
388  rcno->destroyObjectFromDatabase(true);
389  }
390 
391  power -= containerPower;
392  }
