76  results = 0.015f;

79  results = 0.01f;

83  break;
84  case CraftingManager::OK:
85  results = -0.04f;
86  break;
87  case CraftingManager::BARELYSUCCESSFUL:
88  results = -0.07f;

96  }

113 float SharedLabratory::getWeightedValue(ManufactureSchematic* manufactureSchematic, int type) {
114 
115  int nsum = 0;
116  float weightedAverage = 0;
117  int n = 0;
118  int stat = 0;
119 
120  for (int i = 0; i < manufactureSchematic->getSlotCount(); ++i) {
121 
122  Reference<IngredientSlot* > ingredientslot = manufactureSchematic->getSlot(i);
123  Reference<DraftSlot* > draftslot = manufactureSchematic->getDraftSchematic()->getDraftSlot(i);
124 
126  if(!ingredientslot->isResourceSlot())
127  continue;
128 
129  ResourceSlot* resSlot = cast<ResourceSlot*>(ingredientslot.get());
130 
131  if(resSlot == NULL)
132  continue;
133 
134  ManagedReference<ResourceSpawn* > spawn = resSlot->getCurrentSpawn();
135 
136  if (spawn == NULL) {
137  error("Spawn object is null when running getWeightedValue");
138  return 0.0f;
139  }
140 
141  n = draftslot->getQuantity();
142  stat = spawn->getValueOf(type);
143 
144  if (stat != 0) {
145 
146  nsum += n;
147  weightedAverage += (stat * n);
148  }
149  }
150 
151  if (weightedAverage != 0)
152  weightedAverage /= float(nsum);
153 
154  return weightedAverage;
155 }
