75  int doQueueCommand(CreatureObject* creature, const uint64& target, const UnicodeString& arguments) {
76  if (!checkStateMask(creature))
77  return INVALIDSTATE;
78 
79  if (!checkInvalidLocomotions(creature))
80  return INVALIDLOCOMOTION;
81 
82  ManagedReference<Zone*> zone = creature->getZone();
83 
84  if (zone == NULL)
85  return GENERALERROR;
86 
87  ManagedReference<PlanetManager*> planetManager = zone->getPlanetManager();
88 
89  Reference<PlanetTravelPoint*> closestPoint = planetManager->getNearestPlanetTravelPoint(creature, 128.f);
90 
91  // Check to make sure the creature is within range of a PlanetTravelPoint
92  if (closestPoint == NULL) {
93  // Could do @player_structure:boarding_too_far here but this allows you to know in-game that no point was found
94  creature->sendSystemMessage("There is no shuttle nearby.");
95  return GENERALERROR;
96  }
97 
98  ManagedReference<CreatureObject*> shuttle = closestPoint->getShuttle();
99 
100  // Is there a shuttle object related to this point?
101  if (shuttle == NULL) {
102  error("WARNING: Missing a shuttle object:" + closestPoint->toString());
103 
104  // Different error so it's obvious from in-game that the shuttle did not link to this travel point.
105  creature->sendSystemMessage("Shuttle destroyed by terrorists.");
106  return GENERALERROR;
107  }
108 
109  float range = 128.f;
110 
111  if (creature->getParent() == NULL)
112  range = 25.f; //If they are outside, the range is decreased to 25m.
113 
114  if (!shuttle->isInRange(creature, range)) {
115  creature->sendSystemMessage("@player_structure:boarding_too_far"); //You are too far from the shuttle to board.
116  return GENERALERROR;
117  }
118 
119  // Get tickets user has in inventory for this location
120  SortedVector<ManagedReference<TicketObject*> > tickets = findTicketsInInventory(creature, closestPoint);
121 
122  // Do they have any tickets for this location?
123  if (tickets.size() == 0) {
124  creature->sendSystemMessage("@travel:no_ticket"); //You do not have a ticket to board this shuttle.
125  return GENERALERROR;
126  }
127 
128  // Is shuttle ready to board yet?
129  // Shuttle at Theed Spaceport, Naboo should always be available. Even when the shuttle isn't there.
130  if (!closestPoint->isPoint("naboo","Theed Spaceport")){
131  if (!planetManager->checkShuttleStatus(creature, shuttle))
132  return GENERALERROR;
133  }
134 
135  uint64 ticketoid = target;
136 
137  // If only one ticket and no target use the first ticket
138  if (tickets.size() == 1 && ticketoid == 0) {
139  ticketoid = tickets.get(0)->getObjectID();
140  }
141 
142  ManagedReference<TicketObject*> ticketObject = server->getZoneServer()->getObject(ticketoid).castTo<TicketObject*>();
143 
144  //If no ticket was passed as the target, then send the selection box.
145  if (ticketObject == NULL) {
146  sendTicketSelectionBoxTo(creature, tickets);
147  return SUCCESS;
148  }
149 
150  String departurePlanet = ticketObject->getDeparturePlanet();
151  String departurePoint = ticketObject->getDeparturePoint();
152 
153  if (!closestPoint->isPoint(departurePlanet, departurePoint)) {
154  creature->sendSystemMessage("@travel:no_ticket"); //You do not have a ticket to board this shuttle.
155  return GENERALERROR;
156  }
157 
158  String arrivalPlanet = ticketObject->getArrivalPlanet();
159  String arrivalPointName = ticketObject->getArrivalPoint();
160 
161  ManagedReference<Zone*> arrivalZone = server->getZoneServer()->getZone(arrivalPlanet);
162 
163  if (arrivalZone == NULL) {
164  creature->sendSystemMessage("@travel:route_not_available"); //This ticket's route is no longer available.
165  return GENERALERROR;
166  }
167 
168  Reference<PlanetTravelPoint*> arrivalPoint = arrivalZone->getPlanetManager()->getPlanetTravelPoint(arrivalPointName);
169 
170  if (arrivalPoint == NULL || !closestPoint->canTravelTo(arrivalPoint)) {
171  creature->sendSystemMessage("@travel:wrong_shuttle"); //The ticket is not valid for the given shuttle.
172  return GENERALERROR;
173  }
174 
175  ManagedReference<CreatureObject*> targetShuttleObject = arrivalPoint->getShuttle();
176 
177  if (targetShuttleObject != NULL) {
178  ManagedReference<CityRegion*> region = targetShuttleObject->getCityRegion();
179 
180  if (region != NULL) {
181 #ifdef ENABLE_CITY_TRAVEL_LIMIT
182  if (region->getCurrentPlayerCount() >= MAXIMUM_PLAYER_COUNT) {
183  creature->sendSystemMessage("Your destination is currently under maintenance, please try again later.");
184  return GENERALERROR;
185  }
186 #endif
187  if (region->isBanned(creature->getObjectID())) {
188  creature->sendSystemMessage("@city/city:banned_from_that_city"); // You have been banned from traveling to that city by the city militia
189  return GENERALERROR;
190  }
191  }
192  }
193 
194  ManagedReference<CityRegion*> departCity = shuttle->getCityRegion();
195 
196  if (departCity != NULL){
197  if (departCity->isBanned(creature->getObjectID())) {
198  creature->sendSystemMessage("@city/city:youre_city_banned"); // you are banned from this city and may not use any of its public services and structures
199  return GENERALERROR;
200  }
201  }
202 
203  // calculate arrival position
204  Coordinate p;
205  float x;
206  float y;
207 
208  p.initializePosition(arrivalPoint->getArrivalPosition());
209 
210  ManagedReference<CityRegion*> region = targetShuttleObject->getCityRegion();
211 
212  // Randomize the arrival a bit to try and avoid everyone zoning on top of each other
213  // For NPC cities, use the generic method
214  if (region == NULL || region->isClientRegion()) {
215  p.randomizePosition(3);
216 
217  x = p.getPositionX();
218  y = p.getPositionY();
219 
220  } else {
221 
222  // relative orientation of the shuttle
223  float oy = targetShuttleObject->getDirection()->getY();
224  float dirDegrees = (acos(oy) * 180 / M_PI) * 2;
225 
226  // the proper location for arrival is along a 36 degree arc centered on the shuttle's facing axis, between 13 and 16 meters out
227  dirDegrees = dirDegrees - 18 + System::random(36);
228  float dirRadians = dirDegrees * M_PI / 180;
229  float distance = 13 + System::random(3);
230 
231  // update the X & Y positions accordingly
232  x = p.getPositionX() + sin(dirRadians) * distance;
233  y = p.getPositionY() + cos(dirRadians) * distance;
234  }
235 
236  creature->switchZone(arrivalZone->getZoneName(), x, p.getPositionZ(), y, 0);
237 
238  //remove the ticket from inventory and destory it.
239  ManagedReference<SceneObject*> inventory = creature->getSlottedObject("inventory");
240  ticketObject->destroyObjectFromWorld(true);
241 
242  ticketObject->destroyObjectFromDatabase(true);
243  return SUCCESS;
244  }

246 private:
247  SortedVector<ManagedReference<TicketObject*> > findTicketsInInventory(CreatureObject* creature, PlanetTravelPoint* departurePoint) {
248  SortedVector<ManagedReference<TicketObject*> > tickets;
249 
250  ManagedReference<SceneObject*> inventory = creature->getSlottedObject("inventory");
251 
252  if (inventory == NULL)
253  return tickets;
254 
255  int totalObjects = inventory->getContainerObjectsSize();
256 
257  if (totalObjects <= 0)
258  return tickets;
259 
260  for (int i = 0; i < totalObjects; ++i) {
261  ManagedReference<SceneObject*> obj = inventory->getContainerObject(i);
262 
263  if (!obj->isTangibleObject() || !(cast<TangibleObject*>(obj.get()))->isTicketObject())
264  continue;
265 
266  TicketObject* ticket = cast<TicketObject*>( obj.get());
267 
268  //Check to see if the ticket is for this destination
269  if (!departurePoint->isPoint(ticket->getDeparturePlanet(), ticket->getDeparturePoint()))
270  continue;
271 
272  tickets.put(ticket);
273  }
274 
275  return tickets;
276  }

278  void sendTicketSelectionBoxTo(CreatureObject* creature, SortedVector<ManagedReference<TicketObject*> > tickets) {
279  //Make sure it's a player before sending it a sui box...
280  if (!creature->isPlayerCreature())
281  return;
282 
283  CreatureObject* player = cast<CreatureObject*>(creature);
284 
285  ManagedReference<SuiListBox*> suiListBox = new SuiListBox(player, SuiWindowType::TRAVEL_TICKET_SELECTION);
286  suiListBox->setPromptTitle("Select Destination");
287  suiListBox->setPromptText("Select Destination");
288 
289  for (int i = 0; i < tickets.size(); ++i) {
290  ManagedReference<TicketObject*> ticket = tickets.get(i);
291  suiListBox->addMenuItem(ticket->getArrivalPoint(), ticket->getObjectID());
292  }
293 
294  suiListBox->setCallback(new TicketSelectionSuiCallback(server->getZoneServer()));
295 
296  player->getPlayerObject()->addSuiBox(suiListBox);
297  player->sendMessage(suiListBox->generateMessage());
298  }
