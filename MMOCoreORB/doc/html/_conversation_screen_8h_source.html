206  void sendTo(CreatureObject* player, CreatureObject* npc) {
207  NpcConversationMessage* message;
208 
209  if (customText.isEmpty())
210  message = new NpcConversationMessage(player, dialogText);
211  else
212  message = new NpcConversationMessage(player, customText);
213 
214  //Encapsulate this logic better?
215  StringList* optionsList = new StringList(player);
216 
217  for (int i = 0; i < options.size(); ++i) {
218  Reference<ConversationOption*> option = options.get(i);
219 
220  if (option == NULL)
221  continue;
222 
223  optionsList->insertOption(option->getDisplayedName());
224  }
225 
226  player->sendMessage(message);
227  player->sendMessage(optionsList);
228 
229  ConversationScreen* screenToSave = this;
230 
231  //Check if the conversation should be stopped.
232  if (stopConversation) {
233  player->sendMessage(new StopNpcConversation(player, npc->getObjectID()));
234  screenToSave = NULL;
235  }
236 
237  Reference<ConversationSession*> session = player->getActiveSession(SessionFacadeType::CONVERSATION).castTo<ConversationSession* >();
238  if (session != NULL) {
239  session->setLastConversationScreen(screenToSave);
240  }

247  void readObject(LuaObject* luaObject) {
248  screenID = luaObject->getStringField("id");
249  dialogText.setStringId(luaObject->getStringField("leftDialog"));
250  customText = luaObject->getStringField("customDialogText");
251 
252  if (luaObject->getStringField("stopConversation").toLowerCase() == "true") {
253  stopConversation = true;
254  } else {
255  stopConversation = false;
256  }
257 
258  LuaObject optionsTable = luaObject->getObjectField("options");
259 
260  for (int i = 1; i <= optionsTable.getTableSize(); ++i) {
261  lua_rawgeti(luaObject->getLuaState(), -1, i);
262 
263  LuaObject luaObj(luaObject->getLuaState());
264 
265  String optionString = luaObj.getStringAt(1);
266  String linkedId = luaObj.getStringAt(2);
267 
268  Reference<ConversationOption*> option = NULL;
269 
270  if (optionString.beginsWith("@"))
271  option = new ConversationOption(StringIdChatParameter(optionString), linkedId);
272  else
273  option = new ConversationOption(UnicodeString(optionString), linkedId);
274 
275  options.add(option);
276 
277  luaObj.pop();
278  }
279 
280  optionsTable.pop();
281 
282  readOnly = true;
283  }
