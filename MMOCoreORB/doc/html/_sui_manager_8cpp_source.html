107 void SuiManager::handleSuiEventNotification(uint32 boxID, CreatureObject* player, uint32 cancel, Vector<UnicodeString>* args) {
108  uint16 windowType = (uint16) boxID;
109 
110  Locker _lock(player);
111 
112  ManagedReference<PlayerObject*> ghost = player->getPlayerObject();
113 
114  if (ghost == NULL)
115  return;
116 
117  ManagedReference<SuiBox*> suiBox = ghost->getSuiBox(boxID);
118 
119  if (suiBox == NULL)
120  return;
121 
122  //Remove the box from the player, callback can readd it to the player if needed.
123  ghost->removeSuiBox(boxID);
124  suiBox->clearOptions(); //TODO: Eventually SuiBox needs to be cleaned up to not need this.
125 
126  Reference<SuiCallback*> callback = suiBox->getCallback();
127 
128  if (callback != NULL) {
129  callback->run(player, suiBox, cancel, args);
130  return;
131  }
132 
133  StringBuffer msg;
134  msg << "Unknown message callback with SuiWindowType: " << hex << windowType << ". Falling back on old handler system.";
135  //info(msg, true);
136 
137  switch (windowType) {
138  case SuiWindowType::MEDIC_CONSENT:
139  handleConsentBox(player, suiBox, cancel, args);
140  break;
141  case SuiWindowType::DANCING_START:
142  handleStartDancing(player, suiBox, cancel, args);
143  break;
144  case SuiWindowType::DANCING_CHANGE:
145  handleStartDancing(player, suiBox, cancel, args);
146  break;
147  case SuiWindowType::MUSIC_START:
148  handleStartMusic(player, suiBox, cancel, args);
149  break;
150  case SuiWindowType::MUSIC_CHANGE:
151  handleStartMusic(player, suiBox, cancel, args);
152  break;
153  case SuiWindowType::BAND_START:
154  handleStartMusic(player, suiBox, cancel, args);
155  break;
156  case SuiWindowType::BAND_CHANGE:
157  handleStartMusic(player, suiBox, cancel, args);
158  break;
159  case SuiWindowType::BANK_TRANSFER:
160  handleBankTransfer(player, suiBox, cancel, args);
161  break;
162  case SuiWindowType::FISHING:
163  handleFishingAction(player, suiBox, cancel, args);
164  break;
165  case SuiWindowType::CHARACTER_BUILDER_LIST:

369 void SuiManager::handleCharacterBuilderSelectItem(CreatureObject* player, SuiBox* suiBox, uint32 cancel, Vector<UnicodeString>* args) {

371 
372  if (args->size() < 1)
373  return;
374 
375  bool otherPressed = false;
376  int index = 0;
377 
378  if(args->size() > 1) {
379  otherPressed = Bool::valueOf(args->get(0).toString());
380  index = Integer::valueOf(args->get(1).toString());

382  index = Integer::valueOf(args->get(0).toString());

384

391 
392  PlayerObject* ghost = player->getPlayerObject();
393 
394  //If cancel was pressed then we kill the box/menu.
395  if (cancel != 0)
396  return;
397 
398  //Back was pressed. Send the node above it.
399  if (otherPressed) {
400  CharacterBuilderMenuNode* parentNode = currentNode->getParentNode();
401 
402  if(parentNode == NULL)
403  return;
404 
405  cbSui->setCurrentNode(parentNode);
406 
407  ghost->addSuiBox(cbSui);
408  player->sendMessage(cbSui->generateMessage());
409  return;
410  }
411 
412  CharacterBuilderMenuNode* node = currentNode->getChildNodeAt(index);
413 
414  //Node doesn't exist or the index was out of bounds. Should probably resend the menu here.
415  if (node == NULL) {
416  ghost->addSuiBox(cbSui);
417  player->sendMessage(cbSui->generateMessage());
418  return;
419  }
420 
421  if (node->hasChildNodes()) {
422  //If it has child nodes, display them.
423  cbSui->setCurrentNode(node);
424  ghost->addSuiBox(cbSui);
425  player->sendMessage(cbSui->generateMessage());
426  } else {
427

429

432  if (templatePath == "unlearn_all_skills") {

441

444  }
445

447  } else {

449  }
450 
451  } else if (templatePath == "enhance_character") {
452

455
